 92    add_ancillary: bool = True,
 93    deterministic_ids: bool = False,
 94) -> pd.DataFrame:
 95    """
 96    Generate an RCT dataset via CausalDatasetGenerator (thin wrapper), ensuring
 97    randomized treatment independent of X. Keeps y and d as float for ML compatibility.
 98
 99    Notes on effect scale (how `outcome_params` maps into the structural effect):
100      - outcome_type="normal": treatment shifts the mean by (mean["B"] - mean["A"]) on the outcome scale.
101      - outcome_type="binary": treatment shifts the log-odds by (logit(p_B) - logit(p_A)).
102        Reported g0/g1 are on the probability scale.
103      - outcome_type="poisson": treatment shifts the log-mean by log(lam_B / lam_A) where lam = shape * scale.
104        Reported g0/g1 are on the mean (rate) scale.
105
106    Ancillary columns (if add_ancillary=True) are generated from baseline confounders X only
107    (plus RNG), and do not depend on the realized outcome y nor on treatment d. This avoids
108    both outcome leakage and post-treatment adjustment/collider issues in typical ML pipelines.
109    """
110    # RNG for ancillary generation
111    rng = np.random.default_rng(random_state)
112
113    # Validate split
114    split_f = float(split)
115    if not (0.0 < split_f < 1.0):
116        raise ValueError("split must be in (0,1).")
117
118    # Normalize outcome_type
119    ttype = outcome_type.lower()
120    if ttype == "nonnormal":
121        ttype = "poisson"
122    if ttype not in {"binary", "normal", "poisson"}:
123        raise ValueError("outcome_type must be 'binary', 'normal', or 'poisson' (or alias 'nonnormal').")
124
125    # Default outcome_params
126    if outcome_params is None:
127        if ttype == "binary":
128            outcome_params = {"p": {"A": 0.10, "B": 0.12}}
129        elif ttype == "normal":
130            outcome_params = {"mean": {"A": 0.00, "B": 0.20}, "std": 1.0}
131        else:
132            outcome_params = {"shape": 2.0, "scale": {"A": 1.0, "B": 1.1}}
133
134    # Map to natural-scale means
135    if ttype == "binary":
136        pA = float(outcome_params["p"]["A"]); pB = float(outcome_params["p"]["B"])
137        if not (0.0 < pA < 1.0 and 0.0 < pB < 1.0):
138            raise ValueError("For binary outcomes, probabilities must be in (0,1).")
139        mu0_nat, mu1_nat = pA, pB
140    elif ttype == "normal":
141        muA = float(outcome_params["mean"]["A"]); muB = float(outcome_params["mean"]["B"])
142        sd  = float(outcome_params.get("std", 1.0))
143        if not (sd > 0):
144            raise ValueError("For normal outcomes, std must be > 0.")
145        mu0_nat, mu1_nat = muA, muB
146    else:  # poisson
147        shape = float(outcome_params.get("shape", 2.0))
148        scaleA = float(outcome_params["scale"]["A"])
149        scaleB = float(outcome_params["scale"]["B"])
150        lamA = shape * scaleA; lamB = shape * scaleB
151        if not (lamA > 0 and lamB > 0):
152            raise ValueError("For Poisson outcomes, implied rates must be > 0.")
153        mu0_nat, mu1_nat = lamA, lamB
154
155    # Convert to class parameters
156    if ttype == "binary":
157        alpha_y = _logit(mu0_nat)
158        theta = _logit(mu1_nat) - _logit(mu0_nat)
159        outcome_type_cls = "binary"
160        sigma_y = 1.0
161    elif ttype == "normal":
162        alpha_y = mu0_nat
163        theta = mu1_nat - mu0_nat
164        outcome_type_cls = "continuous"
165        sigma_y = sd
166    else:  # poisson
167        alpha_y = float(np.log(mu0_nat))
168        theta = float(np.log(mu1_nat / mu0_nat))
169        outcome_type_cls = "poisson"
170        sigma_y = 1.0
171
172    # Instantiate the unified generator with randomized treatment
173    gen = CausalDatasetGenerator(
174        theta=theta,
175        tau=None,
176        beta_y=None,
177        beta_d=None,
178        g_y=None,
179        g_d=None,
180        alpha_y=alpha_y,
181        alpha_d=_logit(split_f),
182        sigma_y=sigma_y,
183        outcome_type=outcome_type_cls,
184        confounder_specs=confounder_specs,
185        k=int(k),
186        x_sampler=x_sampler,
187        use_copula=False,
188        target_d_rate=None,
189        u_strength_d=0.0,
190        u_strength_y=0.0,
191        seed=random_state,
192    )
193
194    df = gen.generate(n)
195
196    # Ancillary columns (optional)
197    if add_ancillary:
198        df = _add_ancillary_info(df, n, rng, deterministic_ids)
199
200    return df

113        if v is None:
114            return []
115        if isinstance(v, str):
116            return [v]
117        if isinstance(v, list):
118            for item in v:
119                if not isinstance(item, str):
120                    raise TypeError(f"All confounder names must be strings. Found {type(item).__name__}: {item}")
121            seen = set()
122            unique_confs = []
123            for c in v:
124                if c not in seen:
125                    unique_confs.append(c)
126                    seen.add(c)
127            return unique_confs
128        raise TypeError("confounders must be None, a string, or a list of strings.")

131    def _validate_and_normalize(self) -> 'CausalData':
132        """
133        Perform complex validation and normalize the stored DataFrame.
134        """
135        df = self.df
136        treatment = self.treatment_name
137        outcome = self.outcome_name
138        confounders = self.confounders_names
139        user_id = self.user_id_name
140        instrument = self.instrument_name
141
142        # 0. Guard against duplicate column names
143        if df.columns.has_duplicates:
144            dupes = df.columns[df.columns.duplicated()].unique().tolist()
145            raise ValueError(f"DataFrame has duplicate column names: {dupes}. This is not supported.")
146
147        # 1. Disjoint role validation
148        roles = {
149            "outcome": outcome,
150            "treatment": treatment,
151        }
152        if user_id:
153            roles["user_id"] = user_id
154        if instrument:
155            roles["instrument"] = instrument
156
157        # Check for overlaps between primary roles
158        role_names = list(roles.keys())
159        for i, r1 in enumerate(role_names):
160            for r2 in role_names[i+1:]:
161                if roles[r1] == roles[r2]:
162                    raise ValueError(f"Column '{roles[r1]}' cannot be both {r1} and {r2}.")
163
164        overlap = [c for c in confounders if c in set(roles.values())]
165        if overlap:
166            raise ValueError(
167                "confounder columns must be disjoint from treatment/outcome/user_id/instrument; overlapping columns: "
168                + ", ".join(overlap)
169            )
170
171        # 2. Check if all specified columns exist in df
172        all_columns = set(df.columns)
173        for role_name, col in roles.items():
174            if col not in all_columns:
175                raise ValueError(f"Column '{col}' specified as {role_name} does not exist in the DataFrame.")
176        
177        for col in confounders:
178            if col not in all_columns:
179                raise ValueError(f"Column '{col}' specified as confounders does not exist in the DataFrame.")
180
181        # 3. Validate types and constant variance
182        # Outcome
183        if not (pdtypes.is_numeric_dtype(df[outcome]) or pdtypes.is_bool_dtype(df[outcome])):
184            raise ValueError(f"Column '{outcome}' specified as outcome must contain only int, float, or bool values.")
185        if df[outcome].nunique(dropna=False) <= 1:
186            raise ValueError(

188                f"which is not allowed for causal inference."
189            )
190
191        # Treatment

193            raise ValueError(f"Column '{treatment}' specified as treatment must contain only int, float, or bool values.")
194        if df[treatment].nunique(dropna=False) <= 1:

197                f"which is not allowed for causal inference."
198            )
199
200        # Instrument
201        if instrument:
202            if not (pdtypes.is_numeric_dtype(df[instrument]) or pdtypes.is_bool_dtype(df[instrument])):
203                raise ValueError(f"Column '{instrument}' specified as instrument must contain only int, float, or bool values.")
204            if df[instrument].nunique(dropna=False) <= 1:
205                raise ValueError(
206                    f"Column '{instrument}' specified as instrument is constant (has zero variance / single unique value), "
207                    f"which is not allowed for causal inference."
208                )
209
210        # confounders
211        kept_confounders: List[str] = []
212        dropped_constants: List[str] = []

215                raise ValueError(f"Column '{col}' specified as confounders must contain only int, float, or bool values.")
216            
217            if df[col].nunique(dropna=False) <= 1:
218                dropped_constants.append(col)
219            else:
220                kept_confounders.append(col)
221
222        if dropped_constants:
223            warnings.warn(
224                "Dropping constant confounder columns (zero variance): " + ", ".join(dropped_constants),
225                UserWarning,
226                stacklevel=2,
227            )
228        # Update confounders names
229        self.confounders_names = kept_confounders
230        confounders = kept_confounders # Update local variable for next steps
231
232        # 4. Check for NaN values in used columns
233        cols_to_check = [outcome, treatment] + confounders
234        if user_id:
235            cols_to_check.append(user_id)
236        if instrument:
237            cols_to_check.append(instrument)
238        
239        # Unique columns preserving order
240        cols_to_check = list(dict.fromkeys(cols_to_check))
241        
242        if df[cols_to_check].isna().any().any():
243            raise ValueError("DataFrame contains NaN values in used columns, which are not allowed.")
244
245        # 5. Store only the relevant columns and coerce booleans to int8 (except user_id)
246        self.df = df[cols_to_check].copy()
247        for col in self.df.columns:
248            if col != user_id and pdtypes.is_bool_dtype(self.df[col]):
249                self.df[col] = self.df[col].astype("int8")
250            
251            # Final safeguard for numeric types (excluding user_id)
252            if col != user_id:
253                if not pdtypes.is_numeric_dtype(self.df[col]):
254                    raise ValueError(
255                        f"All columns in stored DataFrame must be numeric; column '{col}' has dtype {self.df[col].dtype}."
256                    )
257
258        # 6. Check for duplicate column values and duplicate rows on the normalized subset
259        self._check_duplicate_column_values(self.df)
260        self._check_duplicate_rows(self.df)

160@dataclass(slots=True)
161class CausalDatasetGenerator:
162    """
163    Generate synthetic causal inference datasets with controllable cofounding,
164    treatment prevalence, noise, and (optionally) heterogeneous treatment effects.
165
166    **Data model (high level)**
167
168    - confounders X  R^k are drawn from user-specified distributions.
169    - Binary treatment D is assigned by a logistic model:
170        D ~ Bernoulli( sigmoid(alpha_d + f_d(X) + u_strength_d * U) ),
171      where f_d(X) = X @ beta_d + g_d(X), and U ~ N(0,1) is an optional unobserved confounder.
172    - Outcome Y depends on treatment and confounders with link determined by `outcome_type`:
173        outcome_type = "continuous":
174            Y = alpha_y + f_y(X) + u_strength_y * U + T * tau(X) + ,   ~ N(0, sigma_y^2)
175        outcome_type = "binary":
176            logit P(Y=1|T,X) = alpha_y + f_y(X) + u_strength_y * U + T * tau(X)
177        outcome_type = "poisson":
178            log E[Y|T,X]     = alpha_y + f_y(X) + u_strength_y * U + T * tau(X)
179      where f_y(X) = X @ beta_y + g_y(X), and tau(X) is either constant `theta` or a user function.
180
181    **Returned columns**
182      - y: outcome
183      - d: binary treatment (0/1)
184      - x1..xk (or user-provided names)
185      - m   : true propensity P(T=1 | X)
186      - tau_link: tau(X) on the structural (link) scale (mean-diff for continuous; log-odds for binary; log-mean for Poisson)
187      - g0  : E[Y | X, T=0] on the natural outcome scale
188      - g1  : E[Y | X, T=1] on the natural outcome scale
189      - cate: g1 - g0 (conditional average treatment effect on the natural outcome scale)
190
191    Notes on effect scale:
192      - For "continuous", `theta` (or tau(X)) is an additive mean difference, so `tau_link == cate` (up to noise).
193      - For "binary", tau acts on the *log-odds* scale (log-odds ratio). `cate` is reported as a risk difference
194        (probability scale) and can be heterogeneous even if `tau_link` is constant, because it depends on baseline risk.
195      - For "poisson", tau acts on the *log-mean* scale (log incidence-rate ratio). `cate` is reported on the mean
196        (rate) scale and can be heterogeneous even if `tau_link` is constant, because it depends on the baseline mean.
197
198    Parameters
199    ----------
200    theta : float, default=1.0
201        Constant treatment effect used if `tau` is None.
202
203    tau : callable or None, default=None
204        Function tau(X) -> array-like shape (n,) for heterogeneous effects. Ignored if None.
205
206    beta_y : array-like of shape (k,), optional
207        Linear coefficients of confounders in the outcome baseline f_y(X). Length must match the expanded X after one-hot encoding.
208
209    beta_d : array-like of shape (k,), optional
210        Linear coefficients of confounders in the treatment score f_t(X) (log-odds scale). Length must match the expanded X after one-hot encoding.
211
212    g_y : callable, optional
213        Nonlinear/additive function g_y(X) -> (n,) added to the outcome baseline.
214
215    g_d : callable, optional
216        Nonlinear/additive function g_d(X) -> (n,) added to the treatment score.
217
218    alpha_y : float, default=0.0
219        Outcome intercept (natural scale for continuous; log-odds for binary; log-mean for Poisson).
220
221    alpha_d : float, default=0.0
222        Treatment intercept (log-odds). If `target_d_rate` is set, `alpha_d` is auto-calibrated.
223
224    sigma_y : float, default=1.0
225        Std. dev. of the Gaussian noise for continuous outcomes.
226
227    outcome_type : {"continuous","binary","poisson"}, default="continuous"
228        Outcome family and link as defined above.
229
230    confounder_specs : list of dict, optional
231        Schema for generating confounders. Each spec is one of:
232          {"name": str, "dist": "normal",   "mu": float, "sd": float}
233          {"name": str, "dist": "uniform",  "a": float,  "b": float}
234          {"name": str, "dist": "bernoulli","p": float}
235          {"name": str, "dist": "categorical", "categories": list, "probs": list}
236        - For "categorical", one-hot encoding is produced for all levels except the first.
237
238    k : int, default=5
239        Number of confounders when `confounder_specs` is None. Defaults to independent N(0,1).
240
241    x_sampler : callable, optional
242        Custom sampler (n, k, seed) -> X ndarray of shape (n,k). Overrides `confounder_specs` and `k`.
243
244    target_d_rate : float in (0,1), optional
245        If set, calibrates `alpha_d` via bisection so that the *sample mean* treatment propensity
246        matches `target_d_rate`.
247
248        Important when `u_strength_d != 0`:
249        calibration is performed conditional on the realized draw of the unobserved confounder `U`
250        (i.e., it targets the sample mean of `m_obs = P(D=1|X,U)`). This makes the treatment share
251        match in-sample.
252
253        If you need population calibration (integrating out `U`, i.e. matching the mean of
254        `m = P(D=1|X) = E_U[m_obs]`), you would instead solve for `alpha_d` against the marginal
255        propensity `m` (e.g., using GaussHermite quadrature, analogous to how `m` is computed).
256
257    u_strength_d : float, default=0.0
258        Strength of the unobserved confounder U in treatment assignment.
259
260    u_strength_y : float, default=0.0
261        Strength of the unobserved confounder U in the outcome.
262
263    seed : int, optional
264        Random seed for reproducibility.

267    ----------
268    rng : numpy.random.Generator
269        Internal RNG seeded from `seed`.
270    """
271    # Core knobs
272    theta: float = 1.0                            # constant treatment effect (ATE) if tau is None
273    tau: Optional[Callable[[np.ndarray], np.ndarray]] = None  # heterogeneous effect tau(X) if provided
274
275    # confounder -> outcome/treatment effects
276    beta_y: Optional[np.ndarray] = None           # shape (k,)
277    beta_d: Optional[np.ndarray] = None           # shape (k,)
278    g_y: Optional[Callable[[np.ndarray], np.ndarray]] = None  # nonlinear baseline outcome f_y(X)
279    g_d: Optional[Callable[[np.ndarray], np.ndarray]] = None  # nonlinear treatment score f_d(X)

282    alpha_y: float = 0.0

288    confounder_specs: Optional[List[Dict[str, Any]]] = None   # list of {"name","dist",...}

290    x_sampler: Optional[Callable[[int, int, int], np.ndarray]] = None  # custom sampler (n, k, seed)->X
291    use_copula: bool = False                      # if True and confounder_specs provided, use Gaussian copula
292    copula_corr: Optional[np.ndarray] = None      # correlation matrix for copula (shape dxd where d=len(specs))
293
294    # Practical controls
295    target_d_rate: Optional[float] = None         # e.g., 0.3 -> ~30% treated; solves for alpha_d
296    u_strength_d: float = 0.0                     # unobserved confounder effect on treatment
297    u_strength_y: float = 0.0                     # unobserved confounder effect on outcome

304    def __post_init__(self):
305        self.rng = np.random.default_rng(self.seed)
306        if self.confounder_specs is not None:
307            self.k = len(self.confounder_specs)

311    def _sample_X(self, n: int) -> Tuple[np.ndarray, List[str]]:
312        if self.x_sampler is not None:
313            X = self.x_sampler(n, self.k, self.seed)
314            names = [f"x{i+1}" for i in range(self.k)]
315            return X, names
316
317        if self.confounder_specs is None:
318            # Default: independent standard normals
319            X = self.rng.normal(size=(n, self.k))
320            names = [f"x{i+1}" for i in range(self.k)]
321            return X, names
322
323        # If specs are provided and copula is requested, use Gaussian copula
324        if getattr(self, "use_copula", False):
325            X, names = _gaussian_copula(self.rng, n, self.confounder_specs, getattr(self, "copula_corr", None))
326            self.k = X.shape[1]
327            return X, names
328
329        cols = []
330        names = []
331        for spec in self.confounder_specs:
332            name = spec.get("name") or f"x{len(names)+1}"
333            dist = spec.get("dist", "normal").lower()
334            if dist == "normal":
335                mu = spec.get("mu", 0.0); sd = spec.get("sd", 1.0)
336                col = self.rng.normal(mu, sd, size=n)
337            elif dist == "uniform":
338                a = spec.get("a", 0.0); b = spec.get("b", 1.0)
339                col = self.rng.uniform(a, b, size=n)
340            elif dist == "bernoulli":
341                p = spec.get("p", 0.5)
342                col = self.rng.binomial(1, p, size=n).astype(float)
343            elif dist == "categorical":
344                categories = list(spec.get("categories", [0,1,2]))
345                probs = spec.get("probs", None)
346                if probs is not None:
347                    p = np.asarray(probs, dtype=float)
348                    ps = p / p.sum()
349                else:
350                    ps = None
351                col = self.rng.choice(categories, p=ps, size=n)
352                # one-hot encode (except first level)
353                rest = categories[1:]
354                if len(rest) == 0:
355                    cols.append(np.zeros(n, dtype=float))
356                    names.append(f"{name}__onlylevel")
357                    continue
358                for c in rest:
359                    cols.append((col == c).astype(float))
360                    names.append(f"{name}_{c}")
361                continue
362            else:
363                raise ValueError(f"Unknown dist: {dist}")
364            cols.append(col.astype(float))
365            names.append(name)
366        X = np.column_stack(cols) if cols else np.empty((n,0))
367        self.k = X.shape[1]
368        return X, names

372    def _treatment_score(self, X: np.ndarray, U: np.ndarray) -> np.ndarray:
373        # Ensure numeric, finite arrays
374        Xf = np.asarray(X, dtype=float)
375        # X-driven part of the score
376        score_x = np.zeros(Xf.shape[0], dtype=float)
377        if self.beta_d is not None:
378            bt = np.asarray(self.beta_d, dtype=float)
379            if bt.ndim != 1:
380                bt = bt.reshape(-1)
381            if bt.shape[0] != Xf.shape[1]:
382                raise ValueError(f"beta_d shape {bt.shape} is incompatible with X shape {Xf.shape}")
383            score_x += np.sum(Xf * bt, axis=1)
384        if self.g_d is not None:
385            score_x += np.asarray(self.g_d(Xf), dtype=float)
386        # Scale sharpness to control positivity difficulty
387        s = float(getattr(self, "propensity_sharpness", 1.0))
388        lin = s * score_x
389        # Add unobserved confounder contribution (unscaled)
390        if self.u_strength_d != 0:
391            lin += self.u_strength_d * np.asarray(U, dtype=float)
392        return lin
393
394    def _outcome_location(self, X: np.ndarray, D: np.ndarray, U: np.ndarray, tau_x: np.ndarray) -> np.ndarray:
395        # location on natural scale for continuous; on logit/log scale for binary/poisson
396        Xf = np.asarray(X, dtype=float)
397        Df = np.asarray(D, dtype=float)
398        Uf = np.asarray(U, dtype=float)
399        taux = np.asarray(tau_x, dtype=float)
400        loc = np.full(Xf.shape[0], float(self.alpha_y), dtype=float)
401        if self.beta_y is not None:
402            by = np.asarray(self.beta_y, dtype=float)
403            if by.ndim != 1:
404                by = by.reshape(-1)
405            if by.shape[0] != Xf.shape[1]:
406                raise ValueError(f"beta_y shape {by.shape} is incompatible with X shape {Xf.shape}")
407            loc += np.sum(Xf * by, axis=1)
408        if self.g_y is not None:
409            loc += np.asarray(self.g_y(Xf), dtype=float)
410        if self.u_strength_y != 0:
411            loc += self.u_strength_y * Uf
412        loc += Df * taux
413        return loc
414
415    def _calibrate_alpha_d(self, X: np.ndarray, U: np.ndarray, target: float) -> float:
416        """Calibrate alpha_d so mean propensity ~= target using robust bracketing and bisection."""
417        lo, hi = -50.0, 50.0
418        # Define function whose root we seek
419        def f(a: float) -> float:
420            return float(_sigmoid(a + self._treatment_score(X, U)).mean() - target)
421        flo, fhi = f(lo), f(hi)
422        # If the target is not bracketed, try expanding the bracket a few times
423        if flo * fhi > 0:
424            for scale in (2, 5, 10):
425                lo2, hi2 = -scale * 50.0, scale * 50.0
426                flo2, fhi2 = f(lo2), f(hi2)
427                if flo2 * fhi2 <= 0:
428                    lo, hi = lo2, hi2
429                    flo, fhi = flo2, fhi2
430                    break
431            else:
432                # Fall back to the closer endpoint
433                return lo if abs(flo) < abs(fhi) else hi
434        # Standard bisection with tighter tolerance and bounded iterations
435        for _ in range(80):
436            mid = 0.5 * (lo + hi)
437            fm = f(mid)
438            if abs(fm) < 1e-6:
439                return mid
440            if fm > 0:
441                hi = mid
442            else:
443                lo = mid
444        return 0.5 * (lo + hi)

448    def generate(self, n: int) -> pd.DataFrame:
449        """
450        Draw a synthetic dataset of size `n`.
451        """
452        X, names = self._sample_X(n)
453        U = self.rng.normal(size=n)  # unobserved confounder
454
455        # Treatment assignment
456        if self.target_d_rate is not None:
457            self.alpha_d = self._calibrate_alpha_d(X, U, self.target_d_rate)
458        logits_t = self.alpha_d + self._treatment_score(X, U)
459        m_obs = _sigmoid(logits_t)  # realized propensity including U
460
461        # Marginal m(x) = E[D|X] (integrate out U if it affects treatment)
462        if float(self.u_strength_d) != 0.0:
463            gh_x, gh_w = np.polynomial.hermite.hermgauss(21)
464            gh_w = gh_w / np.sqrt(np.pi)
465            base = self.alpha_d + self._treatment_score(X, np.zeros(n))
466            m = np.sum(_sigmoid(base[:, None] + self.u_strength_d * np.sqrt(2.0) * gh_x[None, :]) * gh_w[None, :], axis=1)
467        else:
468            # Closed form when U doesn't affect D
469            base = self.alpha_d + self._treatment_score(X, np.zeros(n))
470            m = _sigmoid(base)
471
472        D = self.rng.binomial(1, m_obs).astype(float)
473
474        # Treatment effect (constant or heterogeneous)
475        tau_x = (self.tau(X) if self.tau is not None else np.full(n, self.theta)).astype(float)
476
477        # Outcome generation
478        loc = self._outcome_location(X, D, U, tau_x)
479
480        if self.outcome_type == "continuous":
481            Y = loc + self.rng.normal(0, self.sigma_y, size=n)
482        elif self.outcome_type == "binary":
483            # logit: logit P(Y=1|D,X) = loc
484            p = _sigmoid(loc)
485            Y = self.rng.binomial(1, p).astype(float)
486        elif self.outcome_type == "poisson":
487            # log link: log E[Y|D,X] = loc; guard against overflow on link scale
488            loc_c = np.clip(loc, -20, 20)
489            lam = np.exp(loc_c)
490            Y = self.rng.poisson(lam).astype(float)
491        else:
492            raise ValueError("outcome_type must be 'continuous', 'binary', or 'poisson'")
493
494        # Compute oracle g0/g1 on the natural scale, excluding U for continuous, and
495        # marginalizing over U for binary/poisson when u_strength_y != 0.
496        if self.outcome_type == "continuous":
497            # Oracle means exclude U (mean-zero unobserved)
498            g0 = self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n))
499            g1 = self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x)
500
501        elif self.outcome_type == "binary":
502            if float(self.u_strength_y) != 0.0:
503                gh_x, gh_w = np.polynomial.hermite.hermgauss(21)
504                gh_w = gh_w / np.sqrt(np.pi)
505                base0 = self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n))
506                base1 = self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x)
507                Uq = np.sqrt(2.0) * gh_x
508                g0 = np.sum(_sigmoid(base0[:, None] + self.u_strength_y * Uq[None, :]) * gh_w[None, :], axis=1)
509                g1 = np.sum(_sigmoid(base1[:, None] + self.u_strength_y * Uq[None, :]) * gh_w[None, :], axis=1)
510            else:
511                g0 = _sigmoid(self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n)))
512                g1 = _sigmoid(self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x))
513
514        elif self.outcome_type == "poisson":
515            if float(self.u_strength_y) != 0.0:
516                gh_x, gh_w = np.polynomial.hermite.hermgauss(21)
517                gh_w = gh_w / np.sqrt(np.pi)
518                base0 = self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n))
519                base1 = self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x)
520                Uq = np.sqrt(2.0) * gh_x
521                g0 = np.sum(np.exp(np.clip(base0[:, None] + self.u_strength_y * Uq[None, :], -20, 20)) * gh_w[None, :], axis=1)
522                g1 = np.sum(np.exp(np.clip(base1[:, None] + self.u_strength_y * Uq[None, :], -20, 20)) * gh_w[None, :], axis=1)
523            else:
524                g0 = np.exp(np.clip(self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n)), -20, 20))
525                g1 = np.exp(np.clip(self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x), -20, 20))
526        else:
527            raise ValueError("outcome_type must be 'continuous', 'binary', or 'poisson'")
528
529        df = pd.DataFrame({"y": Y, "d": D})
530        for j, name in enumerate(names):
531            df[name] = X[:, j]
532        # Useful ground-truth columns for evaluation (IRM naming)
533        df["m"] = m              # marginal E[D|X]
534        df["m_obs"] = m_obs      # realized sigmoid(alpha_d + score + u*U)
535        df["tau_link"] = tau_x   # structural effect on the link scale
536        df["g0"] = g0
537        df["g1"] = g1
538        df["cate"] = df["g1"] - df["g0"]
539        # Legacy aliases retained for compatibility (no warnings)
540        df["propensity"] = df["m"]
541        df["propensity_obs"] = df["m_obs"]
542        df["mu0"] = df["g0"]
543        df["mu1"] = df["g1"]
544        return df

569    def oracle_nuisance(self, num_quad: int = 21):
570        """
571        Return nuisance functions (m(x), g0(x), g1(x)) compatible with IRM.
572        """
573        # Disallow unobserved cofounding for DML when U affects both T and Y
574        if (getattr(self, "u_strength_d", 0.0) != 0) and (getattr(self, "u_strength_y", 0.0) != 0):
575            raise ValueError(
576                "DML identification fails when U affects both T and Y. "
577                "Use instruments (PLIV-DML) or set one of u_strength_* to 0."
578            )
579
580        # Precompute GH nodes/weights normalized for N(0,1)
581        gh_x, gh_w = np.polynomial.hermite.hermgauss(int(num_quad))
582        gh_w = gh_w / np.sqrt(np.pi)
583
584        def m_of_x(x_row: np.ndarray) -> float:
585            x = np.asarray(x_row, dtype=float).reshape(1, -1)
586            # Base score without U contribution, matching _treatment_score(X, U=0)
587            base = float(self.alpha_d)
588            base += float(self._treatment_score(x, np.zeros(1, dtype=float))[0])
589            ut = float(getattr(self, "u_strength_d", 0.0))
590            if ut == 0.0:
591                return float(_sigmoid(base))
592            # Integrate over U ~ N(0,1) using GaussHermite: U = sqrt(2) * gh_x
593            z = base + ut * np.sqrt(2.0) * gh_x
594            return float(np.sum(_sigmoid(z) * gh_w))
595
596        def g_of_x_d(x_row: np.ndarray, d: int) -> float:
597            x = np.asarray(x_row, dtype=float).reshape(1, -1)
598            if self.tau is None:
599                tau_val = float(self.theta)
600            else:
601                tau_val = float(np.asarray(self.tau(x), dtype=float).reshape(-1)[0])
602            loc = float(self.alpha_y)
603            if self.beta_y is not None:
604                loc += float(np.sum(x * np.asarray(self.beta_y, dtype=float), axis=1)[0])
605            if self.g_y is not None:
606                loc += float(np.asarray(self.g_y(x), dtype=float).reshape(-1)[0])
607            loc += float(d) * tau_val
608
609            if self.outcome_type == "continuous":
610                return float(loc)
611
612            uy = float(getattr(self, "u_strength_y", 0.0))
613            if self.outcome_type == "binary":
614                if uy == 0.0:
615                    return float(_sigmoid(loc))
616                z = loc + uy * np.sqrt(2.0) * gh_x
617                return float(np.sum(_sigmoid(z) * gh_w))
618            if self.outcome_type == "poisson":
619                if uy == 0.0:
620                    return float(np.exp(np.clip(loc, -20.0, 20.0)))
621                z = np.clip(loc + uy * np.sqrt(2.0) * gh_x, -20.0, 20.0)
622                return float(np.sum(np.exp(z) * gh_w))
623            raise ValueError("outcome_type must be 'continuous','binary','poisson'.")
624
625        return (lambda x: m_of_x(x),
626                lambda x: g_of_x_d(x, 0),
627                lambda x: g_of_x_d(x, 1))

Generate synthetic causal inference datasets with controllable cofounding, treatment prevalence, noise, and (optionally) heterogeneous treatment effects.

Data model (high level)

  • confounders X R^k are drawn from user-specified distributions.
  • Binary treatment D is assigned by a logistic model: D ~ Bernoulli( sigmoid(alpha_d + f_d(X) + u_strength_d * U) ), where f_d(X) = X @ beta_d + g_d(X), and U ~ N(0,1) is an optional unobserved confounder.
  • Outcome Y depends on treatment and confounders with link determined by outcome_type: outcome_type = "continuous": Y = alpha_y + f_y(X) + u_strength_y * U + T * tau(X) + , ~ N(0, sigma_y^2) outcome_type = "binary": logit P(Y=1|T,X) = alpha_y + f_y(X) + u_strength_y * U + T * tau(X) outcome_type = "poisson": log E[Y|T,X] = alpha_y + f_y(X) + u_strength_y * U + T * tau(X) where f_y(X) = X @ beta_y + g_y(X), and tau(X) is either constant theta or a user function.

Returned columns

  • y: outcome
  • d: binary treatment (0/1)
  • x1..xk (or user-provided names)
  • m : true propensity P(T=1 | X)
  • tau_link: tau(X) on the structural (link) scale (mean-diff for continuous; log-odds for binary; log-mean for Poisson)
  • g0 : E[Y | X, T=0] on the natural outcome scale
  • g1 : E[Y | X, T=1] on the natural outcome scale
  • cate: g1 - g0 (conditional average treatment effect on the natural outcome scale)

Notes on effect scale:

  • For "continuous", theta (or tau(X)) is an additive mean difference, so tau_link == cate (up to noise).
  • For "binary", tau acts on the log-odds scale (log-odds ratio). cate is reported as a risk difference (probability scale) and can be heterogeneous even if tau_link is constant, because it depends on baseline risk.
  • For "poisson", tau acts on the log-mean scale (log incidence-rate ratio). cate is reported on the mean (rate) scale and can be heterogeneous even if tau_link is constant, because it depends on the baseline mean.
Parameters
  • theta (float, default=1.0): Constant treatment effect used if tau is None.
  • tau (callable or None, default=None): Function tau(X) -> array-like shape (n,) for heterogeneous effects. Ignored if None.
  • beta_y (array-like of shape (k,), optional): Linear coefficients of confounders in the outcome baseline f_y(X). Length must match the expanded X after one-hot encoding.
  • beta_d (array-like of shape (k,), optional): Linear coefficients of confounders in the treatment score f_t(X) (log-odds scale). Length must match the expanded X after one-hot encoding.
  • g_y (callable, optional): Nonlinear/additive function g_y(X) -> (n,) added to the outcome baseline.
  • g_d (callable, optional): Nonlinear/additive function g_d(X) -> (n,) added to the treatment score.
  • alpha_y (float, default=0.0): Outcome intercept (natural scale for continuous; log-odds for binary; log-mean for Poisson).
  • alpha_d (float, default=0.0): Treatment intercept (log-odds). If target_d_rate is set, alpha_d is auto-calibrated.
  • sigma_y (float, default=1.0): Std. dev. of the Gaussian noise for continuous outcomes.
  • outcome_type ({"continuous","binary","poisson"}, default="continuous"): Outcome family and link as defined above.
  • confounder_specs (list of dict, optional): Schema for generating confounders. Each spec is one of: {"name": str, "dist": "normal", "mu": float, "sd": float} {"name": str, "dist": "uniform", "a": float, "b": float} {"name": str, "dist": "bernoulli","p": float} {"name": str, "dist": "categorical", "categories": list, "probs": list}
    • For "categorical", one-hot encoding is produced for all levels except the first.
  • k (int, default=5): Number of confounders when confounder_specs is None. Defaults to independent N(0,1).
  • x_sampler (callable, optional): Custom sampler (n, k, seed) -> X ndarray of shape (n,k). Overrides confounder_specs and k.

  • target_d_rate (float in (0,1), optional): If set, calibrates alpha_d via bisection so that the sample mean treatment propensity matches target_d_rate.

    Important when u_strength_d != 0: calibration is performed conditional on the realized draw of the unobserved confounder U (i.e., it targets the sample mean of m_obs = P(D=1|X,U)). This makes the treatment share match in-sample.

    If you need population calibration (integrating out U, i.e. matching the mean of m = P(D=1|X) = E_U[m_obs]), you would instead solve for alpha_d against the marginal propensity m (e.g., using GaussHermite quadrature, analogous to how m is computed).

  • u_strength_d (float, default=0.0): Strength of the unobserved confounder U in treatment assignment.
  • u_strength_y (float, default=0.0): Strength of the unobserved confounder U in the outcome.
  • seed (int, optional): Random seed for reproducibility.

454

456        if self.target_d_rate is not None:
457            self.alpha_d = self._calibrate_alpha_d(X, U, self.target_d_rate)
458        logits_t = self.alpha_d + self._treatment_score(X, U)
459        m_obs = _sigmoid(logits_t)  # realized propensity including U
460
461        # Marginal m(x) = E[D|X] (integrate out U if it affects treatment)
462        if float(self.u_strength_d) != 0.0:
463            gh_x, gh_w = np.polynomial.hermite.hermgauss(21)
464            gh_w = gh_w / np.sqrt(np.pi)
465            base = self.alpha_d + self._treatment_score(X, np.zeros(n))
466            m = np.sum(_sigmoid(base[:, None] + self.u_strength_d * np.sqrt(2.0) * gh_x[None, :]) * gh_w[None, :], axis=1)
467        else:
468            # Closed form when U doesn't affect D
469            base = self.alpha_d + self._treatment_score(X, np.zeros(n))
470            m = _sigmoid(base)
471
472        D = self.rng.binomial(1, m_obs).astype(float)
473
474        # Treatment effect (constant or heterogeneous)
475        tau_x = (self.tau(X) if self.tau is not None else np.full(n, self.theta)).astype(float)
476
477        # Outcome generation
478        loc = self._outcome_location(X, D, U, tau_x)
479

481            Y = loc + self.rng.normal(0, self.sigma_y, size=n)
482        elif self.outcome_type == "binary":
483            # logit: logit P(Y=1|D,X) = loc
484            p = _sigmoid(loc)
485            Y = self.rng.binomial(1, p).astype(float)
486        elif self.outcome_type == "poisson":
487            # log link: log E[Y|D,X] = loc; guard against overflow on link scale
488            loc_c = np.clip(loc, -20, 20)
489            lam = np.exp(loc_c)
490            Y = self.rng.poisson(lam).astype(float)
491        else:
492            raise ValueError("outcome_type must be 'continuous', 'binary', or 'poisson'")

498            g0 = self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n))
499            g1 = self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x)

504                gh_w = gh_w / np.sqrt(np.pi)
505                base0 = self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n))
506                base1 = self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x)
507                Uq = np.sqrt(2.0) * gh_x
508                g0 = np.sum(_sigmoid(base0[:, None] + self.u_strength_y * Uq[None, :]) * gh_w[None, :], axis=1)
509                g1 = np.sum(_sigmoid(base1[:, None] + self.u_strength_y * Uq[None, :]) * gh_w[None, :], axis=1)
510            else:

512                g1 = _sigmoid(self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x))
513
514        elif self.outcome_type == "poisson":
515            if float(self.u_strength_y) != 0.0:
516                gh_x, gh_w = np.polynomial.hermite.hermgauss(21)
517                gh_w = gh_w / np.sqrt(np.pi)
518                base0 = self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n))
519                base1 = self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x)
520                Uq = np.sqrt(2.0) * gh_x
521                g0 = np.sum(np.exp(np.clip(base0[:, None] + self.u_strength_y * Uq[None, :], -20, 20)) * gh_w[None, :], axis=1)
522                g1 = np.sum(np.exp(np.clip(base1[:, None] + self.u_strength_y * Uq[None, :], -20, 20)) * gh_w[None, :], axis=1)
523            else:
524                g0 = np.exp(np.clip(self._outcome_location(X, np.zeros(n), np.zeros(n), np.zeros(n)), -20, 20))
525                g1 = np.exp(np.clip(self._outcome_location(X, np.ones(n),  np.zeros(n), tau_x), -20, 20))
526        else:
527            raise ValueError("outcome_type must be 'continuous', 'binary', or 'poisson'")
528
529        df = pd.DataFrame({"y": Y, "d": D})
530        for j, name in enumerate(names):
531            df[name] = X[:, j]
532        # Useful ground-truth columns for evaluation (IRM naming)
533        df["m"] = m              # marginal E[D|X]
534        df["m_obs"] = m_obs      # realized sigmoid(alpha_d + score + u*U)
535        df["tau_link"] = tau_x   # structural effect on the link scale
536        df["g0"] = g0
537        df["g1"] = g1
538        df["cate"] = df["g1"] - df["g0"]
539        # Legacy aliases retained for compatibility (no warnings)
540        df["propensity"] = df["m"]
541        df["propensity_obs"] = df["m_obs"]
542        df["mu0"] = df["g0"]
543        df["mu1"] = df["g1"]
544        return df

559            ]
560        elif isinstance(confounders, str):
561            confounder_cols = [confounders]
562        else:

564

567        return CausalData(df=df, treatment='d', outcome='y', confounders=confounder_cols, user_id=user_id)

571        Return nuisance functions (m(x), g0(x), g1(x)) compatible with IRM.
572        """
573        # Disallow unobserved cofounding for DML when U affects both T and Y
574        if (getattr(self, "u_strength_d", 0.0) != 0) and (getattr(self, "u_strength_y", 0.0) != 0):
575            raise ValueError(
576                "DML identification fails when U affects both T and Y. "
577                "Use instruments (PLIV-DML) or set one of u_strength_* to 0."
578            )
579
580        # Precompute GH nodes/weights normalized for N(0,1)
581        gh_x, gh_w = np.polynomial.hermite.hermgauss(int(num_quad))
582        gh_w = gh_w / np.sqrt(np.pi)
583
584        def m_of_x(x_row: np.ndarray) -> float:
585            x = np.asarray(x_row, dtype=float).reshape(1, -1)
586            # Base score without U contribution, matching _treatment_score(X, U=0)
587            base = float(self.alpha_d)
588            base += float(self._treatment_score(x, np.zeros(1, dtype=float))[0])
589            ut = float(getattr(self, "u_strength_d", 0.0))
590            if ut == 0.0:
591                return float(_sigmoid(base))
592            # Integrate over U ~ N(0,1) using GaussHermite: U = sqrt(2) * gh_x
593            z = base + ut * np.sqrt(2.0) * gh_x
594            return float(np.sum(_sigmoid(z) * gh_w))
595
596        def g_of_x_d(x_row: np.ndarray, d: int) -> float:
597            x = np.asarray(x_row, dtype=float).reshape(1, -1)
598            if self.tau is None:
599                tau_val = float(self.theta)
600            else:
601                tau_val = float(np.asarray(self.tau(x), dtype=float).reshape(-1)[0])
602            loc = float(self.alpha_y)
603            if self.beta_y is not None:
604                loc += float(np.sum(x * np.asarray(self.beta_y, dtype=float), axis=1)[0])
605            if self.g_y is not None:
606                loc += float(np.asarray(self.g_y(x), dtype=float).reshape(-1)[0])
607            loc += float(d) * tau_val
608
609            if self.outcome_type == "continuous":
610                return float(loc)
611
612            uy = float(getattr(self, "u_strength_y", 0.0))
613            if self.outcome_type == "binary":
614                if uy == 0.0:
615                    return float(_sigmoid(loc))
616                z = loc + uy * np.sqrt(2.0) * gh_x
617                return float(np.sum(_sigmoid(z) * gh_w))
618            if self.outcome_type == "poisson":
619                if uy == 0.0:
620                    return float(np.exp(np.clip(loc, -20.0, 20.0)))
621                z = np.clip(loc + uy * np.sqrt(2.0) * gh_x, -20.0, 20.0)
622                return float(np.sum(np.exp(z) * gh_w))

624

627                lambda x: g_of_x_d(x, 1))
