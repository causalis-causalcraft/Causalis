 14def ttest(data: CausalData, confidence_level: float = 0.95) -> Dict[str, Any]:
 15    """
 16    Perform a t-test on a CausalData object to compare the outcome variable between
 17    treated (T=1) and control (T=0) groups. Returns differences and confidence intervals.
 18    
 19    Parameters
 20    ----------
 21    data : CausalData
 22        The CausalData object containing treatment and outcome variables.
 23    confidence_level : float, default 0.95
 24        The confidence level for calculating confidence intervals (between 0 and 1).
 25        
 26    Returns
 27    -------
 28    Dict[str, Any]
 29        A dictionary containing:
 30        - p_value: The p-value from the t-test
 31        - absolute_difference: The absolute difference between treatment and control means
 32        - absolute_ci: Tuple of (lower, upper) bounds for the absolute difference confidence interval
 33        - relative_difference: The relative difference (percentage change) between treatment and control means
 34        - relative_ci: Tuple of (lower, upper) bounds for the relative difference confidence interval
 35        
 36    Raises
 37    ------
 38    ValueError
 39        If the CausalData object doesn't have both treatment and outcome variables defined,
 40        or if the treatment variable is not binary.
 41    """
 42    # Basic validation: ensure treatment and outcome are proper Series and non-empty
 43    treatment_var = data.treatment
 44    outcome_var = data.outcome
 45
 46    if not isinstance(treatment_var, pd.Series) or treatment_var.empty:
 47        raise ValueError("causaldata object must have a treatment variable defined")
 48    if not isinstance(outcome_var, pd.Series) or outcome_var.empty:
 49        raise ValueError("causaldata object must have a outcome variable defined")
 50
 51    # Ensure binary treatment
 52    unique_treatments = treatment_var.unique()
 53    if len(unique_treatments) != 2:
 54        raise ValueError("Treatment variable must be binary (have exactly 2 unique values)")
 55
 56    # Build groups by conventional 0/1 coding
 57    control_data = outcome_var[treatment_var == 0]
 58    treatment_data = outcome_var[treatment_var == 1]
 59
 60    # Independent two-sample t-test (pooled variance by default equal_var=True)
 61    t_stat, p_value = stats.ttest_ind(treatment_data, control_data, equal_var=True)
 62
 63    # Means
 64    control_mean = float(control_data.mean())
 65    treatment_mean = float(treatment_data.mean())
 66
 67    # Absolute difference (ATT style: treated - control)
 68    absolute_diff = treatment_mean - control_mean
 69
 70    # Standard error using pooled variance
 71    n1 = int(len(treatment_data))
 72    n2 = int(len(control_data))
 73    s1_squared = float(treatment_data.var(ddof=1))
 74    s2_squared = float(control_data.var(ddof=1))
 75
 76    # Guard against degenerate cases with very small groups
 77    if n1 < 2 or n2 < 2:
 78        raise ValueError("Not enough observations in one of the groups for t-test (need at least 2 per group)")
 79
 80    pooled_var = ((n1 - 1) * s1_squared + (n2 - 1) * s2_squared) / (n1 + n2 - 2)
 81    se_diff = float(np.sqrt(pooled_var * (1 / n1 + 1 / n2)))
 82
 83    # Confidence interval
 84    if not 0 < confidence_level < 1:
 85        raise ValueError("confidence_level must be between 0 and 1 (exclusive)")
 86    alpha = 1 - confidence_level
 87    df = n1 + n2 - 2
 88    t_critical = float(stats.t.ppf(1 - alpha / 2, df))
 89
 90    margin_of_error = t_critical * se_diff
 91    absolute_ci = (absolute_diff - margin_of_error, absolute_diff + margin_of_error)
 92
 93    # Relative difference (%), relative CI via delta method on denominator
 94    if control_mean == 0:
 95        relative_diff = np.inf if absolute_diff > 0 else -np.inf if absolute_diff < 0 else 0.0
 96        relative_ci = (np.nan, np.nan)
 97    else:
 98        relative_diff = (absolute_diff / abs(control_mean)) * 100.0
 99        relative_margin = (margin_of_error / abs(control_mean)) * 100.0
100        relative_ci = (relative_diff - relative_margin, relative_diff + relative_margin)
101
102    return {
103        "p_value": float(p_value),
104        "absolute_difference": float(absolute_diff),
105        "absolute_ci": (float(absolute_ci[0]), float(absolute_ci[1])),
106        "relative_difference": float(relative_diff),
107        "relative_ci": (float(relative_ci[0]), float(relative_ci[1])),
108    }

 19def conversion_z_test(data: CausalData, confidence_level: float = 0.95) -> Dict[str, Any]:
 20    """
 21    Perform a two-proportion z-test on a CausalData object with a binary outcome (conversion).
 22
 23    Parameters
 24    ----------
 25    data : CausalData
 26        The CausalData object containing treatment and outcome variables.
 27    confidence_level : float, default 0.95
 28        The confidence level for calculating confidence intervals (between 0 and 1).
 29
 30    Returns
 31    -------
 32    Dict[str, Any]
 33        A dictionary containing:
 34        - p_value: Two-sided p-value from the z-test
 35        - absolute_difference: Difference in conversion rates (treated - control)
 36        - absolute_ci: Tuple (lower, upper) for the absolute difference CI
 37        - relative_difference: Percentage change relative to control rate
 38        - relative_ci: Tuple (lower, upper) for the relative difference CI
 39
 40    Raises
 41    ------
 42    ValueError
 43        If treatment/outcome are missing, treatment is not binary, outcome is not binary,
 44        groups are empty, or confidence_level is outside (0, 1).
 45    """
 46    # Basic validation
 47    treatment_var = data.treatment
 48    outcome_var = data.outcome
 49
 50    if not isinstance(treatment_var, pd.Series) or treatment_var.empty:
 51        raise ValueError("causaldata object must have a treatment variable defined")
 52    if not isinstance(outcome_var, pd.Series) or outcome_var.empty:
 53        raise ValueError("causaldata object must have a outcome variable defined")
 54
 55    # Treatment must be binary 0/1
 56    tr_unique = treatment_var.unique()
 57    if len(tr_unique) != 2:
 58        raise ValueError("Treatment variable must be binary (have exactly 2 unique values)")
 59
 60    # Target must be binary 0/1 for conversion test
 61    tg_unique = set(pd.Series(outcome_var.unique()).dropna().tolist())
 62    if not tg_unique.issubset({0, 1}):
 63        raise ValueError("Target must be binary (0/1) for conversion_z_test")
 64
 65    # Build groups using 0/1 coding on treatment
 66    control_mask = treatment_var == 0
 67    treat_mask = treatment_var == 1
 68
 69    control = outcome_var[control_mask]
 70    treat = outcome_var[treat_mask]
 71
 72    n0 = int(control.shape[0])
 73    n1 = int(treat.shape[0])
 74    if n0 < 1 or n1 < 1:
 75        raise ValueError("Not enough observations in one of the groups for z-test (need at least 1 per group)")
 76
 77    # Counts of conversions (assumes 0/1 coding for outcome)
 78    x0 = float(control.sum())
 79    x1 = float(treat.sum())
 80
 81    p0 = x0 / n0
 82    p1 = x1 / n1
 83
 84    # Two-proportion z-test (two-sided p-value) using pooled SE under H0 for test statistic
 85    p_pool = (x0 + x1) / (n0 + n1)
 86    se_pooled = float(np.sqrt(p_pool * (1 - p_pool) * (1 / n0 + 1 / n1)))
 87    # Guard against zero standard error (e.g., no variance case)
 88    if se_pooled == 0:
 89        z_stat = 0.0
 90        p_value = 1.0
 91    else:
 92        z_stat = float((p1 - p0) / se_pooled)
 93        p_value = float(2 * (1 - stats.norm.cdf(abs(z_stat))))
 94
 95    # Absolute difference and CI using unpooled (Wald) SE
 96    absolute_diff = float(p1 - p0)
 97
 98    if not 0 < confidence_level < 1:
 99        raise ValueError("confidence_level must be between 0 and 1 (exclusive)")
100    alpha = 1 - confidence_level
101    z_crit = float(stats.norm.ppf(1 - alpha / 2))
102
103    se_unpooled = float(np.sqrt(p1 * (1 - p1) / n1 + p0 * (1 - p0) / n0))
104    margin = z_crit * se_unpooled
105    absolute_ci = (absolute_diff - margin, absolute_diff + margin)
106
107    # Relative difference (%) and CI via scaling by control rate
108    if p0 == 0:
109        relative_diff = np.inf if absolute_diff > 0 else -np.inf if absolute_diff < 0 else 0.0
110        relative_ci = (np.nan, np.nan)
111    else:
112        relative_diff = (absolute_diff / abs(p0)) * 100.0
113        relative_margin = (margin / abs(p0)) * 100.0
114        relative_ci = (relative_diff - relative_margin, relative_diff + relative_margin)
115
116    return {
117        "p_value": float(p_value),
118        "absolute_difference": float(absolute_diff),
119        "absolute_ci": (float(absolute_ci[0]), float(absolute_ci[1])),
120        "relative_difference": float(relative_diff),
121        "relative_ci": (float(relative_ci[0]), float(relative_ci[1])),
122    }

 35    n_simul: int = 10000,
 36) -> Dict[str, Any]:
 37    """
 38    Bootstrap inference for difference in means between treated (T=1) and control (T=0).
 39
 40    Parameters
 41    ----------
 42    data : CausalData
 43        The CausalData object containing treatment and outcome variables.
 44    confidence_level : float, default 0.95
 45        Confidence level for the percentile confidence interval (0 < level < 1).
 46    n_simul : int, default 10000
 47        Number of bootstrap resamples.
 48
 49    Returns
 50    -------
 51    Dict[str, Any]
 52        Dictionary with p_value, absolute_difference, absolute_ci, relative_difference, relative_ci
 53        (matching the structure of inference.atte.ttest).
 54
 55    Raises
 56    ------
 57    ValueError
 58        If inputs are invalid, treatment is not binary, or groups are empty.
 59    """
 60    # Validate inputs
 61    if not 0 < confidence_level < 1:
 62        raise ValueError("confidence_level must be between 0 and 1 (exclusive)")
 63    if not isinstance(n_simul, int) or n_simul <= 0:
 64        raise ValueError("n_simul must be a positive integer")
 65
 66    treatment = data.treatment
 67    outcome = data.outcome
 68
 69    if not isinstance(treatment, pd.Series) or treatment.empty:
 70        raise ValueError("causaldata object must have a treatment variable defined")
 71    if not isinstance(outcome, pd.Series) or outcome.empty:
 72        raise ValueError("causaldata object must have a outcome variable defined")
 73
 74    uniq = treatment.unique()
 75    if len(uniq) != 2:
 76        raise ValueError("Treatment variable must be binary (have exactly 2 unique values)")
 77
 78    control = outcome[treatment == 0]
 79    treated = outcome[treatment == 1]
 80
 81    n0 = int(control.shape[0])
 82    n1 = int(treated.shape[0])
 83    if n0 < 1 or n1 < 1:
 84        raise ValueError("Not enough observations in one of the groups for bootstrap (need at least 1 per group)")
 85
 86    control_mean = float(control.mean())
 87    treated_mean = float(treated.mean())
 88    abs_diff = float(treated_mean - control_mean)
 89
 90    # Prepare for bootstrap: indices for resampling within each group
 91    ctrl_vals = control.to_numpy()
 92    trt_vals = treated.to_numpy()
 93    rng = np.random.default_rng()
 94
 95    # Vectorized bootstrap using random integers for indices
 96    ctrl_idx = rng.integers(0, n0, size=(n_simul, n0))
 97    trt_idx = rng.integers(0, n1, size=(n_simul, n1))
 98
 99    ctrl_boot_means = ctrl_vals[ctrl_idx].mean(axis=1)
100    trt_boot_means = trt_vals[trt_idx].mean(axis=1)
101    boot_diffs = trt_boot_means - ctrl_boot_means
102
103    # Percentile CI for absolute difference
104    alpha = 1 - confidence_level
105    lower = float(np.quantile(boot_diffs, alpha / 2))
106    upper = float(np.quantile(boot_diffs, 1 - alpha / 2))
107    absolute_ci = (lower, upper)
108
109    # p-value using bootstrap SE and normal approximation
110    se_boot = float(np.std(boot_diffs, ddof=1))
111    if se_boot == 0:
112        p_value = 1.0
113    else:
114        z = abs_diff / se_boot
115        p_value = float(2 * (1 - stats.norm.cdf(abs(z))))
116
117    # Relative effects and CI by scaling
118    if control_mean == 0:
119        relative_diff = np.inf if abs_diff > 0 else 0.0 if abs_diff == 0 else -np.inf
120        relative_ci = (np.nan, np.nan)
121    else:
122        relative_diff = (abs_diff / abs(control_mean)) * 100.0
123        rel_lower = (lower / abs(control_mean)) * 100.0
124        rel_upper = (upper / abs(control_mean)) * 100.0
125        relative_ci = (float(rel_lower), float(rel_upper))
126
127    return {
128        "p_value": float(p_value),
129        "absolute_difference": float(abs_diff),
130        "absolute_ci": (float(absolute_ci[0]), float(absolute_ci[1])),
131        "relative_difference": float(relative_diff),
132        "relative_ci": (float(relative_ci[0]), float(relative_ci[1])),
133    }

 28    
 29    Parameters
 30    ----------

 48    Dict[str, Any]
 49        A dictionary containing:
 50        - coefficient: The estimated average treatment effect on the treated
 51        - std_error: The standard error of the estimate
 52        - p_value: The p-value for the null hypothesis that the effect is zero
 53        - confidence_interval: Tuple of (lower, upper) bounds for the confidence interval
 54        - model: The fitted DoubleMLIRM object
 55        
 56    Raises
 57    ------
 58    ValueError
 59        If the causaldata object doesn't have treatment, target, and confounders variables defined,
 60        or if the treatment variable is not binary.
 61        
 62    Examples
 63    --------
 64    >>> from causalis.data import generate_rct_data
 65    >>> from causalis.data import CausalData
 66    >>> from causalis.inference.atte import dml_atte_source
 67    >>> 
 68    >>> # Generate data
 69    >>> df = generate_rct_data()

 71    >>> # Create causaldata object
 72    >>> ck = CausalData(
 73    ...     df=df,
 74    ...     outcome='outcome',
 75    ...     treatment='treatment',
 76    ...     confounders=['age', 'invited_friend']
 77    ... )
 78    >>> 
 79    >>> # Estimate ATT using DoubleML
 80    >>> results = dml_atte_source(ck)
 81    >>> print(f"ATT: {results['coefficient']:.4f}")
 82    >>> print(f"Standard Error: {results['std_error']:.4f}")
 83    >>> print(f"P-value: {results['p_value']:.4f}")
 84    >>> print(f"Confidence Interval: {results['confidence_interval']}")
 85    """
 86    # Validate inputs
 87    if data.treatment is None:
 88        raise ValueError("CausalData object must have a treatment variable defined")
 89    if data.outcome is None:
 90        raise ValueError("CausalData object must have a outcome variable defined")
 91    if not data.confounders:
 92        raise ValueError("CausalData object must have confounders variables defined")
 93    
 94    # Check confidence level
 95    if not 0 < confidence_level < 1:
 96        raise ValueError("confidence_level must be between 0 and 1 (exclusive)")

116    xnames = list(data.confounders)
117

120        df[tname] = t.astype(int)
121    else:

124            raise ValueError(f"Treatment must be binary 0/1 or boolean; found {uniq}.")
125        df[tname] = df[tname].astype(int)
126

128    data_dml = doubleml.DoubleMLData(

130        y_col=yname,
131        d_cols=tname,
132        x_cols=xnames
133    )
134

136    dml_irm_obj = doubleml.DoubleMLIRM(

138        ml_g=ml_g,
139        ml_m=ml_m,
140        n_folds=n_folds,
141        n_rep=n_rep,

143    )

161            ci_lower = float(ci.iloc[0][pct_cols[0]])
162            ci_upper = float(ci.iloc[0][pct_cols[1]])
163        else:

165            ci_upper = float(ci.iloc[0, 1])
166    else:
167        ci_lower = float(ci[0, 0])
168        ci_upper = float(ci[0, 1])
169    
170    # Return results as a dictionary
171    return {

173        "std_error": float(dml_irm_obj.se[0]),
174        "p_value": float(dml_irm_obj.pval[0]),
175        "confidence_interval": (ci_lower, ci_upper),
176        "model": dml_irm_obj
177    }

Parameters
  • data (CausalData): The causaldata object containing treatment, target, and confounders variables.
  • ml_g (estimator, optional): A machine learner implementing fit() and predict() methods for the nuisance function g_0(D,X) = E[Y|X,D]. If None, a CatBoostRegressor configured to use all CPU cores is used.
  • ml_m (classifier, optional): A machine learner implementing fit() and predict_proba() methods for the nuisance function m_0(X) = E[D|X]. If None, a CatBoostClassifier configured to use all CPU cores is used.
  • n_folds (int, default 5): Number of folds for cross-fitting.
  • n_rep (int, default 1): Number of repetitions for the sample splitting.
  • confidence_level (float, default 0.95): The confidence level for calculating confidence intervals (between 0 and 1).
Returns
  • Dict[str, Any]: A dictionary containing:
    • coefficient: The estimated average treatment effect on the treated
    • std_error: The standard error of the estimate
    • p_value: The p-value for the null hypothesis that the effect is zero
    • confidence_interval: Tuple of (lower, upper) bounds for the confidence interval
    • model: The fitted DoubleMLIRM object
Raises
  • ValueError: If the causaldata object doesn't have treatment, target, and confounders variables defined, or if the treatment variable is not binary.
Examples
>>> from causalis.data import generate_rct_data
>>> from causalis.data import CausalData
>>> from causalis.inference.atte import dml_atte_source
>>> 
>>> # Generate data
>>> df = generate_rct_data()
>>> 
>>> # Create causaldata object
>>> ck = CausalData(
...     df=df,
...     outcome='outcome',
...     treatment='treatment',
...     confounders=['age', 'invited_friend']
... )
>>> 
>>> # Estimate ATT using DoubleML
>>> results = dml_atte_source(ck)
>>> print(f"ATT: {results['coefficient']:.4f}")
>>> print(f"Standard Error: {results['std_error']:.4f}")
>>> print(f"P-value: {results['p_value']:.4f}")
>>> print(f"Confidence Interval: {results['confidence_interval']}")
