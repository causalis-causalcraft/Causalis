342def ece_binary(p: np.ndarray, y: np.ndarray, n_bins: int = 10) -> float:
343    """
344    Expected Calibration Error (ECE) for binary labels using equal-width bins on [0,1].
345
346    Parameters
347    ----------
348    p : np.ndarray
349        Predicted probabilities in [0,1]. Will be clipped to [0,1].
350    y : np.ndarray
351        Binary labels {0,1}.
352    n_bins : int, default 10
353        Number of bins.
354
355    Returns
356    -------
357    float
358        ECE value in [0,1].
359    """
360    p = np.asarray(p, dtype=float)
361    y = np.asarray(y, dtype=float)
362    p = np.clip(p, 0.0, 1.0)
363    n_bins = int(n_bins)
364    if p.size == 0 or y.size == 0 or p.size != y.size:
365        return float("nan")
366
367    # Bin indices consistent with equal-width bins on [0,1]
368    b = np.clip((p * n_bins).astype(int), 0, n_bins - 1)
369    sums = np.bincount(b, weights=p, minlength=n_bins)
370    hits = np.bincount(b, weights=y, minlength=n_bins)
371    cnts = np.bincount(b, minlength=n_bins).astype(float)
372    mask = cnts > 0
373    if not np.any(mask):
374        return float("nan")
375    return float(
376        np.average(
377            np.abs(hits[mask] / cnts[mask] - sums[mask] / cnts[mask]),
378            weights=cnts[mask] / cnts[mask].sum(),
