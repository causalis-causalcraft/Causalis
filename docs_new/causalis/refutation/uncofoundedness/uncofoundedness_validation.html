 19def validate_uncofoundedness_balance(
 20    effect_estimation: Dict[str, Any],
 21    *,
 22    threshold: float = 0.1,
 23    normalize: Optional[bool] = None,
 24) -> Dict[str, Any]:
 25    """
 26    Assess covariate balance under the uncofoundedness assumption by computing
 27    standardized mean differences (SMD) both before weighting (raw groups) and
 28    after weighting using the IPW / ATT weights implied by the DML/IRM estimation.
 29
 30    This function expects the result dictionary returned by dml_ate() or dml_att(),
 31    which includes a fitted IRM model and a 'diagnostic_data' entry with the
 32    necessary arrays.
 33
 34    We compute, for each confounder X_j:
 35      - For ATE (weighted): w1 = D/m_hat, w0 = (1-D)/(1-m_hat).
 36      - For ATTE (weighted): Treated weight = 1 for D=1; Control weight w0 = m_hat/(1-m_hat) for D=0.
 37      - If estimation used normalized IPW (normalize_ipw=True), we scale the corresponding
 38        weights by their sample mean (as done in IRM) before computing balance.
 39
 40    The SMD is defined as |mu1 - mu0| / s_pooled, where mu_g are (weighted) means in the
 41    (pseudo-)populations and s_pooled is the square root of the average of the (weighted)
 42    variances in the two groups.
 43
 44    Parameters
 45    ----------
 46    effect_estimation : Dict[str, Any]
 47        Output dict from dml_ate() or dml_att(). Must contain 'model' and 'diagnostic_data'.
 48    threshold : float, default 0.1
 49        Threshold for SMD; values below indicate acceptable balance for most use cases.
 50    normalize : Optional[bool]
 51        Whether to use normalized weights. If None, inferred from effect_estimation['diagnostic_data']['normalize_ipw'].
 52
 53    Returns
 54    -------
 55    Dict[str, Any]
 56        A dictionary with keys:
 57        - 'smd': pd.Series of weighted SMD values indexed by confounder names
 58        - 'smd_unweighted': pd.Series of SMD values computed before weighting (raw groups)
 59        - 'score': 'ATE' or 'ATTE'
 60        - 'normalized': bool used for weighting
 61        - 'threshold': float
 62        - 'pass': bool indicating whether all weighted SMDs are below threshold
 63    """
 64    if not isinstance(effect_estimation, dict):
 65        raise TypeError("effect_estimation must be a dictionary produced by dml_ate() or dml_att()")
 66    if 'model' not in effect_estimation or 'diagnostic_data' not in effect_estimation:
 67        raise ValueError("Input must contain 'model' and 'diagnostic_data' (from dml_ate/dml_att)")
 68
 69    diag = effect_estimation['diagnostic_data']
 70    if not isinstance(diag, dict):
 71        raise ValueError("'diagnostic_data' must be a dict")
 72
 73    # Required arrays
 74    try:
 75        m_hat = np.asarray(diag['m_hat'], dtype=float)
 76        d = np.asarray(diag['d'], dtype=float)
 77        X = np.asarray(diag['x'], dtype=float)
 78        score = str(diag.get('score', '')).upper()
 79        used_norm = bool(diag.get('normalize_ipw', False)) if normalize is None else bool(normalize)
 80    except Exception as e:
 81        raise ValueError(f"diagnostic_data missing required fields: {e}")
 82
 83    if score not in {"ATE", "ATTE"}:
 84        raise ValueError("diagnostic_data['score'] must be 'ATE' or 'ATTE'")
 85    if X.ndim != 2:
 86        raise ValueError("diagnostic_data['x'] must be a 2D array of shape (n, p)")
 87    n, p = X.shape
 88    if m_hat.shape[0] != n or d.shape[0] != n:
 89        raise ValueError("Length of m_hat and d must match number of rows in x")
 90
 91    # Obtain confounder names if available
 92    try:
 93        model = effect_estimation['model']
 94        names = list(getattr(model.data, 'confounders', []))
 95        if not names or len(names) != p:
 96            names = [f"x{j+1}" for j in range(p)]
 97    except Exception:
 98        names = [f"x{j+1}" for j in range(p)]
 99
100    # Build weights per group according to estimand
101    eps = 1e-12
102    m = np.clip(m_hat, eps, 1.0 - eps)
103
104    if score == "ATE":
105        w1 = d / m
106        w0 = (1.0 - d) / (1.0 - m)
107        if used_norm:
108            w1_mean = float(np.mean(w1))
109            w0_mean = float(np.mean(w0))
110            w1 = w1 / (w1_mean if w1_mean != 0 else 1.0)
111            w0 = w0 / (w0_mean if w0_mean != 0 else 1.0)
112    else:  # ATTE
113        w1 = d  # treated weight = 1 for treated, 0 otherwise
114        w0 = (1.0 - d) * (m / (1.0 - m))
115        if used_norm:
116            w1_mean = float(np.mean(w1))
117            w0_mean = float(np.mean(w0))
118            w1 = w1 / (w1_mean if w1_mean != 0 else 1.0)
119            w0 = w0 / (w0_mean if w0_mean != 0 else 1.0)
120
121    # Guard against no-mass groups
122    s1 = float(np.sum(w1))
123    s0 = float(np.sum(w0))
124    if s1 <= 0 or s0 <= 0:
125        raise RuntimeError("Degenerate weights: zero total mass in a pseudo-population")
126
127    # Weighted means and variances (population-style)
128    WX1 = (w1[:, None] * X)
129    WX0 = (w0[:, None] * X)
130    mu1 = WX1.sum(axis=0) / s1
131    mu0 = WX0.sum(axis=0) / s0
132
133    # Weighted variance: E[(X-mu)^2] under weight distribution
134    var1 = (w1[:, None] * (X - mu1) ** 2).sum(axis=0) / s1
135    var0 = (w0[:, None] * (X - mu0) ** 2).sum(axis=0) / s0
136    # Pooled standard deviation directly from variances for clarity
137    s_pooled = np.sqrt(0.5 * (np.maximum(var1, 0.0) + np.maximum(var0, 0.0)))
138
139    # SMD with explicit zero-variance handling (weighted)
140    smd = np.full(p, np.nan, dtype=float)
141    zero_both = (var1 <= 1e-16) & (var0 <= 1e-16)
142    diff = np.abs(mu1 - mu0)
143    mask = (~zero_both) & (s_pooled > 1e-16)
144    smd[mask] = diff[mask] / s_pooled[mask]
145    smd[zero_both & (diff <= 1e-16)] = 0.0
146    smd[zero_both & (diff > 1e-16)] = np.inf
147    smd_s = pd.Series(smd, index=names, dtype=float)
148
149    # --- Unweighted (pre-weighting) SMD using raw treated/control groups ---
150    mask1 = d.astype(bool)
151    mask0 = ~mask1
152    # If any group is empty (shouldn't be in typical settings), guard with nan SMDs
153    if not np.any(mask1) or not np.any(mask0):
154        smd_unw = np.full(p, np.nan, dtype=float)
155    else:
156        X1 = X[mask1]
157        X0 = X[mask0]
158        mu1_u = X1.mean(axis=0)
159        mu0_u = X0.mean(axis=0)
160        # population-style std to mirror weighted computation
161        sd1_u = X1.std(axis=0, ddof=0)
162        sd0_u = X0.std(axis=0, ddof=0)
163        var1_u = sd1_u ** 2
164        var0_u = sd0_u ** 2
165        s_pool_u = np.sqrt(0.5 * (np.maximum(var1_u, 0.0) + np.maximum(var0_u, 0.0)))
166        smd_unw = np.full(p, np.nan, dtype=float)
167        zero_both_u = (var1_u <= 1e-16) & (var0_u <= 1e-16)
168        diff_u = np.abs(mu1_u - mu0_u)
169        mask_u = (~zero_both_u) & (s_pool_u > 1e-16)
170        smd_unw[mask_u] = diff_u[mask_u] / s_pool_u[mask_u]
171        smd_unw[zero_both_u & (diff_u <= 1e-16)] = 0.0
172        smd_unw[zero_both_u & (diff_u > 1e-16)] = np.inf
173    smd_unweighted_s = pd.Series(smd_unw, index=names, dtype=float)
174
175    # Extra quickreport fields
176    smd_max = float(np.nanmax(smd)) if smd.size else float('nan')
177    worst_features = smd_s.sort_values(ascending=False).head(10)
178
179    # Decide pass/fail: ignore non-finite entries; also require low fraction of violations
180    finite_mask = np.isfinite(smd_s.values)
181    if np.any(finite_mask):
182        frac_viol = float(np.mean(smd_s.values[finite_mask] >= float(threshold)))
183        pass_bal = bool(np.all(smd_s.values[finite_mask] < float(threshold)) and (frac_viol < 0.10))
184    else:
185        # If all SMDs are non-finite (e.g., no variation across all features), treat as pass
186        frac_viol = 0.0
187        pass_bal = True
188
189    out = {
190        'smd': smd_s,
191        'smd_unweighted': smd_unweighted_s,
192        'score': score,
193        'normalized': used_norm,
194        'threshold': float(threshold),
195        'pass': pass_bal,
196        'smd_max': smd_max,
197        'worst_features': worst_features,
198    }
199    return out

212def _grade(val: float, warn: float, strong: float, *, larger_is_worse: bool = True) -> str:
213    """Map a scalar to GREEN/YELLOW/RED; NA for nan/inf."""
214    if val is None or (isinstance(val, float) and (np.isnan(val) or np.isinf(val))):
215        return "NA"
216    v = float(val)
217    if larger_is_worse:
218        # smaller values are better
219        return "GREEN" if v < warn else ("YELLOW" if v < strong else "RED")
220    else:
221        # larger values are better
222        return "GREEN" if v >= warn else ("YELLOW" if v >= strong else "RED")

225def _safe_quantiles(a: np.ndarray, qs=(0.5, 0.9, 0.99)) -> _List[float]:
226    a = np.asarray(a, dtype=float)
227    a = a[np.isfinite(a)]
228    if a.size == 0:
229        return [float("nan")] * len(qs)
230    return [float(np.quantile(a, q)) for q in qs]

233def _ks_unweighted(a: np.ndarray, b: np.ndarray) -> float:
234    """Simple unweighted KS distance between two 1D arrays."""
235    a = np.sort(np.asarray(a, dtype=float))
236    b = np.sort(np.asarray(b, dtype=float))
237    if a.size == 0 or b.size == 0:
238        return float("nan")
239    # Evaluate ECDFs on merged grid
240    grid = np.r_[a, b]
241    grid.sort(kind="mergesort")
242    Fa = np.searchsorted(a, grid, side="right") / a.size
243    Fb = np.searchsorted(b, grid, side="right") / b.size
244    return float(np.max(np.abs(Fa - Fb)))

247def _extract_balance_inputs_from_result(
248    res: _Dict[str, _Any] | _Any,
249) -> _Tuple[np.ndarray, np.ndarray, np.ndarray, str, bool, _List[str]]:
250    """
251    Returns X (n,p), m_hat (n,), d (n,), score ('ATE'/'ATTE'), used_norm (bool), names (len p).
252    Accepts:
253      - dict with keys 'model' and 'diagnostic_data' (preferred)
254      - model-like object with .data and cross-fitted m_hat_/predictions
255    """
256    # Result dict path
257    if isinstance(res, dict):
258        diag = res.get("diagnostic_data", {})
259        if isinstance(diag, dict) and all(k in diag for k in ("x", "m_hat", "d")):
260            X = np.asarray(diag["x"], dtype=float)
261            m = np.asarray(diag["m_hat"], dtype=float).ravel()
262            d = np.asarray(diag["d"], dtype=float).ravel()
263            score = str(diag.get("score", "ATE")).upper()
264            used_norm = bool(diag.get("normalize_ipw", False))
265            names = diag.get("x_names")
266            if not names or len(names) != X.shape[1]:
267                # try model data for names
268                model = res.get("model", None)
269                if model is not None and getattr(model, "data", None) is not None:
270                    try:
271                        names = list(getattr(model.data, "confounders", []))
272                    except Exception:
273                        names = None
274            if not names or len(names) != X.shape[1]:
275                names = [f"x{j+1}" for j in range(X.shape[1])]
276            return X, m, d, ("ATTE" if "ATT" in score else "ATE"), used_norm, names
277
278        # Fall through to model extraction if diag missing
279        res = res.get("model", res)
280
281    # Model-like path
282    model = res
283    data_obj = getattr(model, "data", None)
284    if data_obj is None or not hasattr(data_obj, "get_df"):
285        raise ValueError("Could not extract data arrays. Provide `res` with diagnostic_data or a model with .data.get_df().")
286
287    df = data_obj.get_df()
288    confs = list(getattr(data_obj, "confounders", [])) or []
289    if not confs:
290        raise ValueError("CausalData must include confounders to compute balance (X).")
291    X = df[confs].to_numpy(dtype=float)
292    names = confs
293
294    # m_hat
295    if hasattr(model, "m_hat_") and getattr(model, "m_hat_", None) is not None:
296        m = np.asarray(model.m_hat_, dtype=float).ravel()
297    else:
298        preds = getattr(model, "predictions", None)
299        if isinstance(preds, dict) and "ml_m" in preds:
300            m = np.asarray(preds["ml_m"], dtype=float).ravel()
301        else:
302            raise AttributeError("Could not locate propensity predictions (m_hat_ or predictions['ml_m']).")
303
304    # d
305    tname = getattr(getattr(data_obj, "treatment", None), "name", None) or getattr(data_obj, "_treatment", "D")
306    d = df[tname].to_numpy(dtype=float).ravel()
307
308    # score & normalization
309    sc = getattr(model, "score", None) or getattr(model, "_score", "ATE")
310    score = "ATTE" if "ATT" in str(sc).upper() else "ATE"
311    used_norm = bool(getattr(model, "normalize_ipw", False))
312    return X, m, d, score, used_norm, names

317def _balance_smd(
318    X: np.ndarray,
319    d: np.ndarray,
320    m_hat: np.ndarray,
321    *,
322    score: str,
323    normalize: bool,
324    threshold: float,
325) -> _Dict[str, _Any]:
326    """
327    Compute weighted/unweighted SMDs + quick summaries using ATE/ATTE implied weights.
328    """
329    n, p = X.shape
330    eps = 1e-12
331    m = np.clip(np.asarray(m_hat, dtype=float).ravel(), eps, 1.0 - eps)
332    d = np.asarray(d, dtype=float).ravel()
333    score_u = str(score).upper()
334
335    if score_u == "ATE":
336        w1 = d / m
337        w0 = (1.0 - d) / (1.0 - m)
338    else:  # ATTE
339        w1 = d
340        w0 = (1.0 - d) * (m / (1.0 - m))
341
342    if normalize:
343        w1 /= float(np.mean(w1)) if np.mean(w1) != 0 else 1.0
344        w0 /= float(np.mean(w0)) if np.mean(w0) != 0 else 1.0
345
346    s1 = float(np.sum(w1))
347    s0 = float(np.sum(w0))
348    if s1 <= 0 or s0 <= 0:
349        raise RuntimeError("Degenerate weights: zero total mass in a pseudo-population.")
350
351    # weighted means/vars
352    mu1 = (w1[:, None] * X).sum(axis=0) / s1
353    mu0 = (w0[:, None] * X).sum(axis=0) / s0
354    var1 = (w1[:, None] * (X - mu1) ** 2).sum(axis=0) / s1
355    var0 = (w0[:, None] * (X - mu0) ** 2).sum(axis=0) / s0
356    s_pool = np.sqrt(0.5 * (np.maximum(var1, 0.0) + np.maximum(var0, 0.0)))
357    smd_w = np.full(p, np.nan, dtype=float)
358    zero_both = (var1 <= 1e-16) & (var0 <= 1e-16)
359    diff = np.abs(mu1 - mu0)
360    mask = (~zero_both) & (s_pool > 1e-16)
361    smd_w[mask] = diff[mask] / s_pool[mask]
362    smd_w[zero_both & (diff <= 1e-16)] = 0.0
363    smd_w[zero_both & (diff > 1e-16)] = np.inf
364
365    # raw SMD
366    mask1 = d.astype(bool)
367    mask0 = ~mask1
368    if not np.any(mask1) or not np.any(mask0):
369        smd_u = np.full(p, np.nan)
370    else:
371        mu1_u = X[mask1].mean(axis=0)
372        mu0_u = X[mask0].mean(axis=0)
373        sd1_u = X[mask1].std(axis=0, ddof=0)
374        sd0_u = X[mask0].std(axis=0, ddof=0)
375        var1_u = sd1_u ** 2
376        var0_u = sd0_u ** 2
377        s_pool_u = np.sqrt(0.5 * (np.maximum(var1_u, 0.0) + np.maximum(var0_u, 0.0)))
378        smd_u = np.full(p, np.nan)
379        zero_both_u = (var1_u <= 1e-16) & (var0_u <= 1e-16)
380        diff_u = np.abs(mu1_u - mu0_u)
381        mask_u = (~zero_both_u) & (s_pool_u > 1e-16)
382        smd_u[mask_u] = diff_u[mask_u] / s_pool_u[mask_u]
383        smd_u[zero_both_u & (diff_u <= 1e-16)] = 0.0
384        smd_u[zero_both_u & (diff_u > 1e-16)] = np.inf
385
386    # summaries
387    finite = np.isfinite(smd_w)
388    smd_max = float(np.nanmax(smd_w)) if np.any(finite) else float("nan")
389    frac_viol = float(np.mean(smd_w[finite] >= float(threshold))) if np.any(finite) else 0.0
390
391    return {
392        "smd_weighted": smd_w,
393        "smd_unweighted": smd_u,
394        "smd_max": smd_max,
395        "frac_violations": frac_viol,
396        "weights": (w1, w0),
397        "mass": (s1, s0),
398    }

403def run_uncofoundedness_diagnostics(
404    *,
405    res: _Dict[str, _Any] | _Any = None,
406    X: _Optional[np.ndarray] = None,
407    d: _Optional[np.ndarray] = None,
408    m_hat: _Optional[np.ndarray] = None,
409    names: _Optional[_List[str]] = None,
410    score: _Optional[str] = None,
411    normalize: _Optional[bool] = None,
412    threshold: float = 0.10,
413    eps_overlap: float = 0.01,
414    return_summary: bool = True,
415) -> _Dict[str, _Any]:
416    """
417    Uncofoundedness diagnostics focused on balance (SMD).
418
419    Inputs:
420      - Either a result/model via `res`, or raw arrays X, d, m_hat (+ optional names, score, normalize).
421
422    Returns a dictionary:
423      {
424        "params": {"score", "normalize", "smd_threshold"},
425        "balance": {"smd", "smd_unweighted", "smd_max", "frac_violations", "pass", "worst_features"},
426        "flags": {"balance_max_smd", "balance_violations"},
427        "overall_flag": max severity across balance flags,
428        "summary": pd.DataFrame with balance rows only
429      }
430    """
431    # ---- Resolve inputs ----
432    if (X is None or d is None or m_hat is None) and res is None:
433        raise ValueError("Pass either (X, d, m_hat) or `res` with diagnostic_data/model.")
434
435    if X is None or d is None or m_hat is None:
436        X, m_hat, d, score_auto, used_norm_auto, names_auto = _extract_balance_inputs_from_result(res)
437        if score is None:
438            score = score_auto
439        if normalize is None:
440            normalize = used_norm_auto
441        if names is None:
442            names = names_auto
443    else:
444        if score is None:
445            score = "ATE"
446        if normalize is None:
447            normalize = False
448        if names is None:
449            names = [f"x{j+1}" for j in range(X.shape[1])]
450
451    score_u = str(score).upper()
452    used_norm = bool(normalize)
453    X = np.asarray(X, dtype=float)
454    d = np.asarray(d, dtype=float).ravel()
455    m_hat = np.asarray(m_hat, dtype=float).ravel()
456    n, p = X.shape
457    if m_hat.size != n or d.size != n:
458        raise ValueError("X, d, m_hat must have matching length n.")
459
460    # ---- Balance (SMD only) ----
461    bal = _balance_smd(X, d, m_hat, score=score_u, normalize=used_norm, threshold=threshold)
462
463    smd_w = pd.Series(bal["smd_weighted"], index=names, dtype=float, name="SMD_weighted")
464    smd_u = pd.Series(bal["smd_unweighted"], index=names, dtype=float, name="SMD_unweighted")
465    worst = smd_w.sort_values(ascending=False).head(10)
466
467    frac_viol = float(bal["frac_violations"]) if np.isfinite(bal["frac_violations"]) else 0.0
468    pass_bal = bool(np.all(smd_w[np.isfinite(smd_w)] < float(threshold)) and (frac_viol < 0.10)) if np.any(np.isfinite(smd_w)) else True
469
470    balance_block = {
471        "smd": smd_w,
472        "smd_unweighted": smd_u,
473        "smd_max": float(bal["smd_max"]),
474        "frac_violations": frac_viol,
475        "pass": pass_bal,
476        "worst_features": worst,
477    }
478
479
480    # ---- Balance-only flags & thresholds ----
481    def _grade_balance(val: float, warn: float, strong: float) -> str:
482        if val is None or (isinstance(val, float) and (np.isnan(val) or np.isinf(val))):
483            return "NA"
484        v = float(val)
485        return "GREEN" if v < warn else ("YELLOW" if v < strong else "RED")
486
487    smd_warn = float(threshold)
488    smd_strong = float(threshold) * 2.0  # conventional 0.10/0.20 if threshold=0.10
489    viol_frac_warn = 0.10
490    viol_frac_strong = 0.25
491
492    balance_flags = {
493        "balance_max_smd": _grade_balance(balance_block["smd_max"], smd_warn, smd_strong),
494        "balance_violations": _grade_balance(balance_block["frac_violations"], viol_frac_warn, viol_frac_strong),
495    }
496
497    # ---- Overall flag severity (balance only) ----
498    order = {"GREEN": 0, "YELLOW": 1, "RED": 2, "NA": -1}
499    worst_flag = max((order.get(f, -1) for f in balance_flags.values()), default=-1)
500    inv = {v: k for k, v in order.items()}
501    overall_flag = inv.get(worst_flag, "NA")
502
503    # ---- Params ----
504    params = {
505        "score": score_u,
506        "normalize": used_norm,
507        "smd_threshold": float(threshold),
508    }
509
510    report: _Dict[str, _Any] = {
511        "params": params,
512        "balance": balance_block,
513        "flags": balance_flags,
514        "overall_flag": overall_flag,
515        "meta": {"n": int(n), "p": int(p)},
516    }
517

 67    if 'model' not in effect_estimation or 'diagnostic_data' not in effect_estimation:
 68        raise ValueError("Input must contain 'model' and 'diagnostic_data' (from dml_ate/dml_att)")
 69
 70    diag = effect_estimation['diagnostic_data']
 71    if not isinstance(diag, dict):
 72        raise ValueError("'diagnostic_data' must be a dict")
 73
 74    # Required arrays
 75    try:
 76        m_hat = np.asarray(diag['m_hat'], dtype=float)
 77        d = np.asarray(diag['d'], dtype=float)
 78        X = np.asarray(diag['x'], dtype=float)
 79        score = str(diag.get('score', '')).upper()
 80        used_norm = bool(diag.get('normalize_ipw', False)) if normalize is None else bool(normalize)
 81    except Exception as e:
 82        raise ValueError(f"diagnostic_data missing required fields: {e}")
 83
 84    if score not in {"ATE", "ATTE"}:
 85        raise ValueError("diagnostic_data['score'] must be 'ATE' or 'ATTE'")
 86    if X.ndim != 2:
 87        raise ValueError("diagnostic_data['x'] must be a 2D array of shape (n, p)")
 88    n, p = X.shape
 89    if m_hat.shape[0] != n or d.shape[0] != n:
 90        raise ValueError("Length of m_hat and d must match number of rows in x")
 91
 92    # Obtain confounder names if available
 93    try:
 94        model = effect_estimation['model']
 95        names = list(getattr(model.data, 'confounders', []))
 96        if not names or len(names) != p:
 97            names = [f"x{j+1}" for j in range(p)]
 98    except Exception:
 99        names = [f"x{j+1}" for j in range(p)]
100
101    # Build weights per group according to estimand
102    eps = 1e-12
103    m = np.clip(m_hat, eps, 1.0 - eps)
104
105    if score == "ATE":
106        w1 = d / m
107        w0 = (1.0 - d) / (1.0 - m)
108        if used_norm:
109            w1_mean = float(np.mean(w1))
110            w0_mean = float(np.mean(w0))
111            w1 = w1 / (w1_mean if w1_mean != 0 else 1.0)
112            w0 = w0 / (w0_mean if w0_mean != 0 else 1.0)
113    else:  # ATTE
114        w1 = d  # treated weight = 1 for treated, 0 otherwise
115        w0 = (1.0 - d) * (m / (1.0 - m))
116        if used_norm:
117            w1_mean = float(np.mean(w1))
118            w0_mean = float(np.mean(w0))
119            w1 = w1 / (w1_mean if w1_mean != 0 else 1.0)
120            w0 = w0 / (w0_mean if w0_mean != 0 else 1.0)
121
122    # Guard against no-mass groups
123    s1 = float(np.sum(w1))
124    s0 = float(np.sum(w0))
125    if s1 <= 0 or s0 <= 0:
126        raise RuntimeError("Degenerate weights: zero total mass in a pseudo-population")
127
128    # Weighted means and variances (population-style)
129    WX1 = (w1[:, None] * X)
130    WX0 = (w0[:, None] * X)
131    mu1 = WX1.sum(axis=0) / s1
132    mu0 = WX0.sum(axis=0) / s0
133
134    # Weighted variance: E[(X-mu)^2] under weight distribution
135    var1 = (w1[:, None] * (X - mu1) ** 2).sum(axis=0) / s1
136    var0 = (w0[:, None] * (X - mu0) ** 2).sum(axis=0) / s0
137    # Pooled standard deviation directly from variances for clarity
138    s_pooled = np.sqrt(0.5 * (np.maximum(var1, 0.0) + np.maximum(var0, 0.0)))
139
140    # SMD with explicit zero-variance handling (weighted)
141    smd = np.full(p, np.nan, dtype=float)
142    zero_both = (var1 <= 1e-16) & (var0 <= 1e-16)
143    diff = np.abs(mu1 - mu0)
144    mask = (~zero_both) & (s_pooled > 1e-16)
145    smd[mask] = diff[mask] / s_pooled[mask]
146    smd[zero_both & (diff <= 1e-16)] = 0.0
147    smd[zero_both & (diff > 1e-16)] = np.inf
148    smd_s = pd.Series(smd, index=names, dtype=float)
149
150    # --- Unweighted (pre-weighting) SMD using raw treated/control groups ---
151    mask1 = d.astype(bool)
152    mask0 = ~mask1
153    # If any group is empty (shouldn't be in typical settings), guard with nan SMDs
154    if not np.any(mask1) or not np.any(mask0):
155        smd_unw = np.full(p, np.nan, dtype=float)
156    else:
157        X1 = X[mask1]
158        X0 = X[mask0]
159        mu1_u = X1.mean(axis=0)
160        mu0_u = X0.mean(axis=0)
161        # population-style std to mirror weighted computation
162        sd1_u = X1.std(axis=0, ddof=0)
163        sd0_u = X0.std(axis=0, ddof=0)
164        var1_u = sd1_u ** 2
165        var0_u = sd0_u ** 2
166        s_pool_u = np.sqrt(0.5 * (np.maximum(var1_u, 0.0) + np.maximum(var0_u, 0.0)))
167        smd_unw = np.full(p, np.nan, dtype=float)
168        zero_both_u = (var1_u <= 1e-16) & (var0_u <= 1e-16)
169        diff_u = np.abs(mu1_u - mu0_u)
170        mask_u = (~zero_both_u) & (s_pool_u > 1e-16)
171        smd_unw[mask_u] = diff_u[mask_u] / s_pool_u[mask_u]
172        smd_unw[zero_both_u & (diff_u <= 1e-16)] = 0.0
173        smd_unw[zero_both_u & (diff_u > 1e-16)] = np.inf
174    smd_unweighted_s = pd.Series(smd_unw, index=names, dtype=float)
175
176    # Extra quickreport fields
177    smd_max = float(np.nanmax(smd)) if smd.size else float('nan')
178    worst_features = smd_s.sort_values(ascending=False).head(10)
179
180    # Decide pass/fail: ignore non-finite entries; also require low fraction of violations
181    finite_mask = np.isfinite(smd_s.values)
182    if np.any(finite_mask):
183        frac_viol = float(np.mean(smd_s.values[finite_mask] >= float(threshold)))
184        pass_bal = bool(np.all(smd_s.values[finite_mask] < float(threshold)) and (frac_viol < 0.10))
185    else:
186        # If all SMDs are non-finite (e.g., no variation across all features), treat as pass
187        frac_viol = 0.0
188        pass_bal = True
189
190    out = {
191        'smd': smd_s,
192        'smd_unweighted': smd_unweighted_s,
193        'score': score,
194        'normalized': used_norm,
195        'threshold': float(threshold),
196        'pass': pass_bal,
197        'smd_max': smd_max,
198        'worst_features': worst_features,
199    }
