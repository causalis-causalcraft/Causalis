causalis.refutation.score.score_validation

AIPW orthogonality diagnostics for IRM-based estimators.

This module implements comprehensive orthogonality diagnostics for AIPW/IRM-based estimators like dml_ate and dml_att to validate the key assumptions required for valid causal inference. Based on the efficient influence function (EIF) framework.

Key diagnostics implemented:

  • Out-of-sample moment check (non-tautological)
  • Orthogonality (Gateaux derivative) tests
  • Influence diagnostics

  26def aipw_score_ate(y: np.ndarray, d: np.ndarray, g0: np.ndarray, g1: np.ndarray, m: np.ndarray, theta: float, trimming_threshold: float = 0.01) -> np.ndarray:
  27    """
  28    Efficient influence function (EIF) for ATE.
  29    Uses IRM naming: g0,g1 are outcome regressions E[Y|X,D=0/1], m is propensity P(D=1|X).
  30    """
  31    m = np.clip(m, trimming_threshold, 1 - trimming_threshold)
  32    return (g1 - g0) + d * (y - g1) / m - (1 - d) * (y - g0) / (1 - m) - theta

  35def aipw_score_atte(y: np.ndarray, d: np.ndarray, g0: np.ndarray, g1: np.ndarray, m: np.ndarray, theta: float, p_treated: Optional[float] = None, trimming_threshold: float = 0.01) -> np.ndarray:
  36    """
  37    Efficient influence function (EIF) for ATTE under IRM/AIPW.
  38
  39    ψ_ATTE(W; θ, η) = [ D*(Y - g0(X) - θ)  -  (1-D)*{ m(X)/(1-m(X)) }*(Y - g0(X)) ] / E[D]
  40
  41    Notes:
  42      - Matches DoubleML's `score='ATTE'` (weights ω=D/E[D], \bar{ω}=m(X)/E[D]).
  43      - g1 enters only via θ; ∂ψ/∂g1 = 0.
  44    """
  45    m = np.clip(m, trimming_threshold, 1 - trimming_threshold)
  46    if p_treated is None:
  47        p_treated = float(np.mean(d))
  48    gamma = m / (1.0 - m)
  49    num = d * (y - g0 - theta) - (1.0 - d) * gamma * (y - g0)
  50    return num / (p_treated + 1e-12)

  55def extract_nuisances(model, test_indices: Optional[np.ndarray] = None) -> Tuple[np.ndarray, np.ndarray, np.ndarray]:
  56    """
  57    Extract cross-fitted nuisance predictions from an IRM-like model or a compatible dummy.
  58
  59    Tries several backends for robustness:
  60      1) IRM attributes: m_hat_, g0_hat_, g1_hat_
  61      2) model.predictions dict with keys: 'ml_m','ml_g0','ml_g1'
  62      3) Direct attributes: ml_m, ml_g0, ml_g1
  63
  64    Parameters
  65    ----------
  66    model : object
  67        Fitted internal IRM estimator (causalis.inference.estimators.IRM) or a compatible dummy model
  68    test_indices : np.ndarray, optional
  69        If provided, extract predictions only for these indices
  70
  71    Returns
  72    -------
  73    Tuple[np.ndarray, np.ndarray, np.ndarray]
  74        (m, g0, g1) where:
  75        - m: propensity scores P(D=1|X)
  76        - g0: outcome predictions E[Y|X,D=0]
  77        - g1: outcome predictions E[Y|X,D=1]
  78    """
  79    m = g0 = g1 = None
  80
  81    # 1) Preferred IRM attributes
  82    if hasattr(model, "m_hat_") and getattr(model, "m_hat_", None) is not None:
  83        m = np.asarray(getattr(model, "m_hat_"), dtype=float).ravel()
  84        g0 = np.asarray(getattr(model, "g0_hat_"), dtype=float).ravel()
  85        g1 = np.asarray(getattr(model, "g1_hat_"), dtype=float).ravel()
  86    else:
  87        # 2) Predictions dict backend
  88        preds = getattr(model, "predictions", None)
  89        if isinstance(preds, dict) and all(k in preds for k in ("ml_m", "ml_g0", "ml_g1")):
  90            m = np.asarray(preds["ml_m"], dtype=float).ravel()
  91            g0 = np.asarray(preds["ml_g0"], dtype=float).ravel()
  92            g1 = np.asarray(preds["ml_g1"], dtype=float).ravel()
  93        else:
  94            # 3) Direct attributes with same names
  95            if all(hasattr(model, nm) for nm in ("ml_m", "ml_g0", "ml_g1")):
  96                m = np.asarray(getattr(model, "ml_m"), dtype=float).ravel()
  97                g0 = np.asarray(getattr(model, "ml_g0"), dtype=float).ravel()
  98                g1 = np.asarray(getattr(model, "ml_g1"), dtype=float).ravel()
  99
 100    if m is None or g0 is None or g1 is None:
 101        raise AttributeError("IRM model-compatible nuisances not found. Expected m_hat_/g0_hat_/g1_hat_ or predictions['ml_*'].")
 102
 103    if test_indices is not None:
 104        m = m[test_indices]
 105        g0 = g0[test_indices]
 106        g1 = g1[test_indices]
 107
 108    return m, g0, g1

 113def oos_moment_check_with_fold_nuisances(
 114    fold_thetas: List[float],
 115    fold_indices: List[np.ndarray],
 116    fold_nuisances: List[Tuple[np.ndarray, np.ndarray, np.ndarray]],
 117    y: np.ndarray,
 118    d: np.ndarray,
 119    score_fn: Optional[Callable[[np.ndarray, np.ndarray, np.ndarray, np.ndarray, np.ndarray, float], np.ndarray]] = None,
 120) -> Tuple[pd.DataFrame, float]:
 121    """
 122    Out-of-sample moment check using fold-specific nuisances to avoid tautological results.
 123    
 124    For each fold k, evaluates the AIPW score using θ fitted on other folds and 
 125    nuisance predictions from the fold-specific model, then tests if the combined 
 126    moment condition holds.
 127    
 128    Parameters
 129    ----------
 130    fold_thetas : List[float]
 131        Treatment effects estimated excluding each fold
 132    fold_indices : List[np.ndarray] 
 133        Indices for each fold
 134    fold_nuisances : List[Tuple[np.ndarray, np.ndarray, np.ndarray]]
 135        Fold-specific nuisance predictions (m, g0, g1) for each fold
 136    y, d : np.ndarray
 137        Observed outcomes and treatments
 138        
 139    Returns
 140    -------
 141    Tuple[pd.DataFrame, float]
 142        Fold-wise results and combined t-statistic
 143    """
 144    rows = []
 145    for k, (idx, nuisances) in enumerate(zip(fold_indices, fold_nuisances)):
 146        th = fold_thetas[k]
 147        m_fold, g0_fold, g1_fold = nuisances
 148        
 149        # Compute scores using fold-specific theta and nuisances
 150        if score_fn is None:
 151            psi = aipw_score_ate(y[idx], d[idx], g0_fold, g1_fold, m_fold, th)
 152        else:
 153            psi = score_fn(y[idx], d[idx], g0_fold, g1_fold, m_fold, th)
 154        rows.append({"fold": k, "n": len(idx), "psi_mean": psi.mean(), "psi_var": psi.var(ddof=1)})
 155    
 156    df = pd.DataFrame(rows)
 157    num = (df["n"] * df["psi_mean"]).sum()
 158    den = np.sqrt((df["n"] * df["psi_var"]).sum())
 159    tstat = num / den if den > 0 else 0.0
 160    return df, float(tstat)

 209def orthogonality_derivatives(X_basis: np.ndarray, y: np.ndarray, d: np.ndarray, 
 210                            g0: np.ndarray, g1: np.ndarray, m: np.ndarray, trimming_threshold: float = 0.01) -> pd.DataFrame:
 211    """
 212    Compute orthogonality (Gateaux derivative) tests for nuisance functions (ATE case).
 213    Uses IRM naming: g0,g1 outcomes; m propensity.
 214    """
 215    n, B = X_basis.shape
 216    
 217    # Clip propensity scores to avoid division by zero and extreme weights
 218    m_clipped = np.clip(m, trimming_threshold, 1 - trimming_threshold)
 219    
 220    # g1 direction: ∂_{g1} φ̄[h1] = (1/n)∑ h1(Xi)(1 - Di/mi)
 221    dg1_terms = X_basis * (1 - d / m_clipped)[:, None]
 222    dg1 = dg1_terms.mean(axis=0)
 223    dg1_se = dg1_terms.std(axis=0, ddof=1) / np.sqrt(n)
 224    
 225    # g0 direction: ∂_{g0} φ̄[h0] = (1/n)∑ h0(Xi)((1-Di)/(1-mi) - 1)
 226    dg0_terms = X_basis * ((1 - d) / (1 - m_clipped) - 1)[:, None]
 227    dg0 = dg0_terms.mean(axis=0)  
 228    dg0_se = dg0_terms.std(axis=0, ddof=1) / np.sqrt(n)
 229    
 230    # m direction: ∂_m φ̄[s] = -(1/n)∑ s(Xi)[Di(Yi-g1i)/mi² + (1-Di)(Yi-g0i)/(1-mi)²]
 231    m_summand = (d * (y - g1) / m_clipped**2 + (1 - d) * (y - g0) / (1 - m_clipped)**2)
 232    dm_terms = -X_basis * m_summand[:, None]
 233    dm = dm_terms.mean(axis=0)
 234    dm_se = dm_terms.std(axis=0, ddof=1) / np.sqrt(n)
 235    
 236    out = pd.DataFrame({
 237        "basis": np.arange(B),
 238        "d_g1": dg1, "se_g1": dg1_se, "t_g1": dg1 / np.maximum(dg1_se, 1e-12),
 239        "d_g0": dg0, "se_g0": dg0_se, "t_g0": dg0 / np.maximum(dg0_se, 1e-12),
 240        "d_m": dm,  "se_m": dm_se,  "t_m":  dm / np.maximum(dm_se, 1e-12),
 241    })
 242    return out

 245def influence_summary(y: np.ndarray, d: np.ndarray, g0: np.ndarray, g1: np.ndarray, 
 246                     m: np.ndarray, theta_hat: float, k: int = 10, score: str = "ATE", trimming_threshold: float = 0.01) -> Dict[str, Any]:
 247    """
 248    Compute influence diagnostics showing where uncertainty comes from.
 249    
 250    Parameters
 251    ----------
 252    y, d, g0, g1, m : np.ndarray
 253        Data arrays
 254    theta_hat : float
 255        Estimated treatment effect
 256    k : int, default 10
 257        Number of top influential observations to return
 258        
 259    Returns
 260    -------
 261    Dict[str, Any]
 262        Influence diagnostics including SE, heavy-tail metrics, and top-k cases
 263    """
 264    score_u = str(score).upper()
 265    if score_u == "ATE":
 266        psi = aipw_score_ate(y, d, g0, g1, m, theta_hat, trimming_threshold=trimming_threshold)
 267    elif score_u in ("ATTE", "ATT"):
 268        psi = aipw_score_atte(y, d, g0, g1, m, theta_hat, p_treated=float(np.mean(d)), trimming_threshold=trimming_threshold)
 269    else:
 270        raise ValueError("score must be 'ATE' or 'ATTE'")
 271    se = psi.std(ddof=1) / np.sqrt(len(psi))
 272    idx = np.argsort(-np.abs(psi))[:k]
 273    
 274    top = pd.DataFrame({
 275        "i": idx, 
 276        "psi": psi[idx], 
 277        "m": m[idx],
 278        "res_t": d[idx] * (y[idx] - g1[idx]),
 279        "res_c": (1 - d[idx]) * (y[idx] - g0[idx])
 280    })
 281    
 282    return {
 283        "se_plugin": float(se),
 284        "kurtosis": float(((psi - psi.mean())**4).mean() / (psi.var(ddof=1)**2 + 1e-12)),
 285        "p99_over_med": float(np.quantile(np.abs(psi), 0.99) / (np.median(np.abs(psi)) + 1e-12)),
 286        "top_influential": top
 287    }

 290def refute_irm_orthogonality(
 291    inference_fn: Callable[..., Dict[str, Any]],
 292    data: CausalData,
 293    trim_propensity: Tuple[float, float] = (0.02, 0.98),
 294    n_basis_funcs: Optional[int] = None,
 295    n_folds_oos: int = 4,
 296    score: Optional[str] = None,
 297    trimming_threshold: float = 0.01,
 298    strict_oos: bool = True,
 299    **inference_kwargs,
 300) -> Dict[str, Any]:
 301    """
 302    Comprehensive AIPW orthogonality diagnostics for IRM estimators.
 303    
 304    Implements three key diagnostic approaches based on the efficient influence function (EIF):
 305    1. Out-of-sample moment check (non-tautological)
 306    2. Orthogonality (Gateaux derivative) tests  
 307    3. Influence diagnostics
 308    
 309    Parameters
 310    ----------
 311    inference_fn : Callable
 312        The inference function (dml_ate or dml_att)
 313    data : CausalData
 314        The causal data object
 315    trim_propensity : Tuple[float, float], default (0.02, 0.98)
 316        Propensity score trimming bounds (min, max) to avoid extreme weights
 317    n_basis_funcs : Optional[int], default None (len(confounders)+1)
 318        Number of basis functions for orthogonality derivative tests (constant + covariates).
 319        If None, defaults to the number of confounders in `data` plus 1 for the constant term.
 320    n_folds_oos : int, default 5
 321        Number of folds for out-of-sample moment check
 322    **inference_kwargs : dict
 323        Additional arguments passed to inference_fn
 324        
 325    Returns
 326    -------
 327    Dict[str, Any]
 328        Dictionary containing:
 329        - oos_moment_test: Out-of-sample moment condition results
 330        - orthogonality_derivatives: Gateaux derivative test results
 331        - influence_diagnostics: Influence function diagnostics
 332        - theta: Original treatment effect estimate
 333        - trimmed_diagnostics: Results on trimmed sample
 334        - overall_assessment: Summary diagnostic assessment
 335        
 336    Examples
 337    --------
 338    >>> from causalis.refutation.orthogonality import refute_irm_orthogonality
 339    >>> from causalis.inference.ate.dml_ate import dml_ate
 340    >>> 
 341    >>> # Comprehensive orthogonality check
 342    >>> ortho_results = refute_irm_orthogonality(dml_ate, causal_data)
 343    >>> 
 344    >>> # Check key diagnostics
 345    >>> print(f"OOS moment t-stat: {ortho_results['oos_moment_test']['tstat']:.3f}")
 346    >>> print(f"Assessment: {ortho_results['overall_assessment']}")
 347    """
 348    # Run inference to get fitted IRM model
 349    result = inference_fn(data, **inference_kwargs)
 350    
 351    if 'model' not in result:
 352        raise ValueError("Inference function must return a dictionary with 'model' key")
 353    
 354    dml_model = result['model']
 355    # Prefer wrapper-reported coefficient; fallback to IRM attributes if missing
 356    if 'coefficient' in result:
 357        theta_hat = float(result['coefficient'])
 358    else:
 359        theta_hat = float(getattr(dml_model, 'coef_', [np.nan])[0])
 360    
 361    # Extract cross-fitted predictions from IRM model using robust function
 362    m_propensity, g0_outcomes, g1_outcomes = extract_nuisances(dml_model)
 363    
 364    # Get observed data
 365    y = data.outcome.values.astype(float)
 366    d = data.treatment.values.astype(float)
 367
 368    # Determine score
 369    score_u = str(score).upper() if score is not None else "AUTO"
 370    if score_u == "AUTO":
 371        score_attr = getattr(dml_model, 'score', None)
 372        if score_attr is None:
 373            score_attr = getattr(dml_model, '_score', 'ATE')
 374        score_str = str(score_attr).upper()
 375        score_u = 'ATTE' if 'ATT' in score_str else 'ATE'
 376    
 377    # Build score function
 378    if score_u == 'ATE':
 379        def score_fn(y_, d_, g0_, g1_, m_, th_):
 380            return aipw_score_ate(y_, d_, g0_, g1_, m_, th_, trimming_threshold=trimming_threshold)
 381    elif score_u in ('ATTE', 'ATT'):
 382        p_treated = float(np.mean(d))
 383        def score_fn(y_, d_, g0_, g1_, m_, th_):
 384            return aipw_score_atte(y_, d_, g0_, g1_, m_, th_, p_treated=p_treated, trimming_threshold=trimming_threshold)
 385    else:
 386        raise ValueError("score must be 'ATE' or 'ATTE'")
 387    
 388    # Apply propensity trimming
 389    trim_min, trim_max = trim_propensity
 390    trim_mask = (m_propensity >= trim_min) & (m_propensity <= trim_max)
 391    n_trimmed = int(np.sum(~trim_mask))
 392    
 393    # Create trimmed arrays
 394    y_trim = y[trim_mask]
 395    d_trim = d[trim_mask]
 396    m_trim = m_propensity[trim_mask]
 397    g0_trim = g0_outcomes[trim_mask]
 398    g1_trim = g1_outcomes[trim_mask]
 399    
 400    # === 1. OUT-OF-SAMPLE MOMENT CHECK ===
 401    # Prefer fast path using cached ψ_a, ψ_b and training folds from the fitted IRM model
 402    use_fast = hasattr(dml_model, "psi_a_") and hasattr(dml_model, "psi_b_") and getattr(dml_model, "folds_", None) is not None
 403    if use_fast:
 404        folds_arr = np.asarray(dml_model.folds_, dtype=int)
 405        K = int(folds_arr.max() + 1) if folds_arr.size > 0 else 0
 406        fold_indices = [np.where(folds_arr == k)[0] for k in range(K)]
 407        # Compute both fold-aggregated and strict t-stats using only ψ components
 408        oos_df, oos_tstat, tstat_strict = oos_moment_check_from_psi(
 409            np.asarray(dml_model.psi_a_, dtype=float),
 410            np.asarray(dml_model.psi_b_, dtype=float),
 411            fold_indices,
 412            strict=True,
 413        )
 414        oos_pvalue = float(2 * (1 - stats.norm.cdf(abs(oos_tstat))))
 415        pvalue_strict = float(2 * (1 - stats.norm.cdf(abs(tstat_strict)))) if tstat_strict is not None else float("nan")
 416        strict_applied = bool(strict_oos)
 417        main_tstat = float(tstat_strict if strict_applied else oos_tstat)
 418        main_pvalue = float(2 * (1 - stats.norm.cdf(abs(main_tstat))))
 419    else:
 420        # Fallback: legacy slow path with K refits and score recomputations
 421        kf = KFold(n_splits=n_folds_oos, shuffle=True, random_state=42)
 422        fold_thetas: List[float] = []
 423        fold_indices: List[np.ndarray] = []
 424        fold_nuisances: List[Tuple[np.ndarray, np.ndarray, np.ndarray]] = []
 425
 426        df_full = data.get_df()
 427        for train_idx, test_idx in kf.split(df_full):
 428            # Create fold-specific data using public APIs
 429            train_data = CausalData(
 430                df=df_full.iloc[train_idx].copy(),
 431                treatment=data.treatment.name,
 432                outcome=data.outcome.name,
 433                confounders=list(data.confounders) if data.confounders else None
 434            )
 435            # Fit model on training fold to get theta
 436            fold_result = inference_fn(train_data, **inference_kwargs)
 437            fold_thetas.append(float(fold_result['coefficient']))
 438            fold_indices.append(test_idx)
 439
 440            # Use full-model cross-fitted nuisances indexed by test fold (fast OOS)
 441            m_fold = m_propensity[test_idx]
 442            g0_fold = g0_outcomes[test_idx]
 443            g1_fold = g1_outcomes[test_idx]
 444            fold_nuisances.append((m_fold, g0_fold, g1_fold))
 445
 446        # Run out-of-sample moment check with fold-specific nuisances (fold-aggregated)
 447        oos_df, oos_tstat = oos_moment_check_with_fold_nuisances(
 448            fold_thetas, fold_indices, fold_nuisances, y, d, score_fn=score_fn
 449        )
 450        oos_pvalue = float(2 * (1 - stats.norm.cdf(abs(oos_tstat))))
 451
 452        # Strict aggregation over all held-out psi without creating a giant array
 453        total_n = 0
 454        total_sum = 0.0
 455        total_sumsq = 0.0
 456        for k, (idx, (m_fold, g0_fold, g1_fold)) in enumerate(zip(fold_indices, fold_nuisances)):
 457            th = fold_thetas[k]
 458            psi_k = score_fn(y[idx], d[idx], g0_fold, g1_fold, m_fold, th)
 459            n_k = psi_k.size
 460            s_k = float(psi_k.sum())
 461            ss_k = float((psi_k * psi_k).sum())
 462            total_n += n_k
 463            total_sum += s_k
 464            total_sumsq += ss_k
 465        if total_n > 1:
 466            mean_all = total_sum / total_n
 467            var_all = (total_sumsq - total_n * (mean_all ** 2)) / (total_n - 1)
 468            se_all = float(np.sqrt(max(var_all, 0.0) / total_n))
 469            tstat_strict = float(mean_all / se_all) if se_all > 0 else 0.0
 470        else:
 471            tstat_strict = 0.0
 472        pvalue_strict = float(2 * (1 - stats.norm.cdf(abs(tstat_strict))))
 473
 474        # Choose which t-stat to report as main
 475        strict_applied = bool(strict_oos)
 476        main_tstat = float(tstat_strict if strict_applied else oos_tstat)
 477        main_pvalue = float(pvalue_strict if strict_applied else oos_pvalue)
 478    
 479    # === 2. ORTHOGONALITY DERIVATIVE TESTS ===
 480    # Create basis functions: constant + first few standardized confounders
 481    if data.confounders is None or len(data.confounders) == 0:
 482        raise ValueError("CausalData object must have confounders defined for orthogonality diagnostics")
 483
 484    # If n_basis_funcs is not provided, default to (number of confounders + 1) for the constant term
 485    if n_basis_funcs is None:
 486        n_basis_funcs = len(data.confounders) + 1
 487    
 488    # Ensure confounders are represented as a float ndarray to avoid object-dtype reductions
 489    df_conf = data.get_df()[list(data.confounders)]
 490    X = df_conf.to_numpy(dtype=float)
 491    n_covs = min(max(n_basis_funcs - 1, 0), X.shape[1])  # -1 for constant term
 492    
 493    if n_covs > 0:
 494        X_selected = X[:, :n_covs]
 495        X_std = (X_selected - np.mean(X_selected, axis=0)) / (np.std(X_selected, axis=0) + 1e-8)
 496        X_basis = np.c_[np.ones(len(X)), X_std]  # Constant + standardized covariates
 497    else:
 498        X_basis = np.ones((len(X), 1))  # Only constant term
 499    
 500    if score_u == 'ATE':
 501        ortho_derivs_full = orthogonality_derivatives(X_basis, y, d, g0_outcomes, g1_outcomes, m_propensity, trimming_threshold=trimming_threshold)
 502        X_basis_trim = X_basis[trim_mask]
 503        ortho_derivs_trim = orthogonality_derivatives(X_basis_trim, y_trim, d_trim, g0_trim, g1_trim, m_trim, trimming_threshold=trimming_threshold)
 504        problematic_derivs_full = ortho_derivs_full[(np.abs(ortho_derivs_full['t_g1']) > 2) | (np.abs(ortho_derivs_full['t_g0']) > 2) | (np.abs(ortho_derivs_full['t_m']) > 2)]
 505        problematic_derivs_trim = ortho_derivs_trim[(np.abs(ortho_derivs_trim['t_g1']) > 2) | (np.abs(ortho_derivs_trim['t_g0']) > 2) | (np.abs(ortho_derivs_trim['t_m']) > 2)]
 506        derivs_interpretation = 'Large |t| (>2) indicate calibration issues'
 507        derivs_full_ok = len(problematic_derivs_full) == 0
 508        derivs_trim_ok = len(problematic_derivs_trim) == 0
 509    else:  # ATTE / ATT
 510        p_treated_full = float(np.mean(d))
 511        ortho_derivs_full = orthogonality_derivatives_atte(X_basis, y, d, g0_outcomes, m_propensity, p_treated_full, trimming_threshold=trimming_threshold)
 512        X_basis_trim = X_basis[trim_mask]
 513        p_treated_trim = float(np.mean(d_trim))
 514        ortho_derivs_trim = orthogonality_derivatives_atte(X_basis_trim, y_trim, d_trim, g0_trim, m_trim, p_treated_trim, trimming_threshold=trimming_threshold)
 515        # Backward-compatible alias columns for expected legacy names
 516        try:
 517            ortho_derivs_full = ortho_derivs_full.copy()
 518            ortho_derivs_full["d_g"] = ortho_derivs_full.get("d_g0", np.nan)
 519            ortho_derivs_full["t_g"] = ortho_derivs_full.get("t_g0", np.nan)
 520            ortho_derivs_full["d_m0"] = ortho_derivs_full.get("d_m", np.nan)
 521            ortho_derivs_full["t_m0"] = ortho_derivs_full.get("t_m", np.nan)
 522            ortho_derivs_trim = ortho_derivs_trim.copy()
 523            ortho_derivs_trim["d_g"] = ortho_derivs_trim.get("d_g0", np.nan)
 524            ortho_derivs_trim["t_g"] = ortho_derivs_trim.get("t_g0", np.nan)
 525            ortho_derivs_trim["d_m0"] = ortho_derivs_trim.get("d_m", np.nan)
 526            ortho_derivs_trim["t_m0"] = ortho_derivs_trim.get("t_m", np.nan)
 527        except Exception:
 528            pass
 529        problematic_derivs_full = ortho_derivs_full[(np.abs(ortho_derivs_full['t_g0']) > 2) | (np.abs(ortho_derivs_full['t_m']) > 2)]
 530        problematic_derivs_trim = ortho_derivs_trim[(np.abs(ortho_derivs_trim['t_g0']) > 2) | (np.abs(ortho_derivs_trim['t_m']) > 2)]
 531        derivs_interpretation = 'ATTE: check g0 & m only; large |t| (>2) => calibration issues'
 532        derivs_full_ok = len(problematic_derivs_full) == 0
 533        derivs_trim_ok = len(problematic_derivs_trim) == 0
 534    
 535    # === 3. INFLUENCE DIAGNOSTICS ===
 536    influence_full = influence_summary(y, d, g0_outcomes, g1_outcomes, m_propensity, theta_hat, score=score_u, trimming_threshold=trimming_threshold)
 537
 538    # Re-estimate theta on the trimmed sample for fair trimmed influence diagnostics
 539    theta_hat_trim = theta_hat
 540    try:
 541        df_full_local = data.get_df()
 542        data_trim_obj = CausalData(
 543            df=df_full_local.loc[trim_mask].copy(),
 544            treatment=data.treatment.name,
 545            outcome=data.outcome.name,
 546            confounders=list(data.confounders) if data.confounders else None
 547        )
 548        res_trim = inference_fn(data_trim_obj, **inference_kwargs)
 549        theta_hat_trim = float(res_trim.get('coefficient', theta_hat))
 550    except Exception:
 551        pass
 552
 553    influence_trim = influence_summary(y_trim, d_trim, g0_trim, g1_trim, m_trim, theta_hat_trim, score=score_u, trimming_threshold=trimming_threshold)
 554    
 555    # === DIAGNOSTIC ASSESSMENT ===
 556    model_se = result.get('std_error')
 557    if model_se is not None:
 558        se_reasonable = abs(influence_full['se_plugin'] - model_se) < 2 * model_se
 559    else:
 560        se_reasonable = False
 561    
 562    conditions = {
 563        'oos_moment_ok': abs(main_tstat) < 2.0,
 564        'derivs_full_ok': derivs_full_ok,
 565        'derivs_trim_ok': derivs_trim_ok,
 566        'se_reasonable': se_reasonable,
 567        'no_extreme_influence': influence_full['p99_over_med'] < 10,
 568        'trimming_reasonable': n_trimmed < 0.1 * len(y)
 569    }
 570    
 571    n_passed = sum(bool(v) for v in conditions.values())
 572    if n_passed >= 5:
 573        overall_assessment = "PASS: Strong evidence for orthogonality"
 574    elif n_passed >= 4:
 575        overall_assessment = "CAUTION: Most conditions satisfied"
 576    elif n_passed >= 3:
 577        overall_assessment = "WARNING: Several orthogonality violations"
 578    else:
 579        overall_assessment = "FAIL: Major orthogonality violations detected"
 580
 581    # ATTE-specific overlap and trim-sensitivity
 582    overlap_atte = None
 583    trim_curve_atte = None
 584    if score_u in ('ATTE', 'ATT'):
 585        overlap_atte = overlap_diagnostics_atte(m_propensity, d, eps_list=[0.95, 0.97, 0.98, 0.99])
 586        try:
 587            trim_curve_atte = trim_sensitivity_curve_atte(
 588                inference_fn, data, m_propensity, d,
 589                thresholds=np.linspace(0.90, 0.995, 12),
 590                **inference_kwargs
 591            )
 592        except Exception as _:
 593            trim_curve_atte = None
 594    
 595    # p-treated parameterization
 596    params_extra = {}
 597    if score_u in ('ATTE','ATT'):
 598        p_full = float(np.mean(d))
 599        p_trim = float(np.mean(d_trim)) if len(d_trim) > 0 else float('nan')
 600        params_extra = {
 601            'p_treated': p_full,
 602            'p_treated_full': p_full,
 603            'p_treated_trim': p_trim,
 604        }
 605    
 606    return {
 607        'theta': float(theta_hat),
 608        'params': {
 609            'score': score_u,
 610            'trimming_threshold': trimming_threshold,
 611            'strict_oos_requested': bool(strict_oos),
 612            'strict_oos_applied': bool(strict_applied),
 613            **params_extra,
 614        },
 615        'oos_moment_test': {
 616            'fold_results': oos_df,
 617            'tstat': float(main_tstat),
 618            'pvalue': float(main_pvalue),
 619            'tstat_fold_agg': float(oos_tstat),
 620            'pvalue_fold_agg': float(oos_pvalue),
 621            'tstat_strict': float(tstat_strict),
 622            'pvalue_strict': float(pvalue_strict),
 623            'aggregation': 'strict' if strict_applied else 'fold',
 624            'interpretation': 'Should be ≈ 0 if moment condition holds'
 625        },
 626        'orthogonality_derivatives': {
 627            'full_sample': ortho_derivs_full,
 628            'trimmed_sample': ortho_derivs_trim,
 629            'problematic_full': problematic_derivs_full,
 630            'problematic_trimmed': problematic_derivs_trim,
 631            'interpretation': derivs_interpretation
 632        },
 633        'influence_diagnostics': {
 634            'full_sample': influence_full,
 635            'trimmed_sample': influence_trim,
 636            'interpretation': 'Heavy tails or extreme kurtosis suggest instability'
 637        },
 638        'overlap_atte': overlap_atte,
 639        'robustness': {
 640            'trim_curve_atte': trim_curve_atte,
 641            'interpretation': 'ATTE typically more sensitive to trimming near m→1 (controls).'
 642        },
 643        'trimming_info': {
 644            'bounds': trim_propensity,
 645            'n_trimmed': int(n_trimmed),
 646            'pct_trimmed': float(n_trimmed / len(y) * 100.0)
 647        },
 648        'diagnostic_conditions': conditions,
 649        'overall_assessment': overall_assessment
 650    }

 653def orthogonality_derivatives_atte(
 654    X_basis: np.ndarray, y: np.ndarray, d: np.ndarray,
 655    g0: np.ndarray, m: np.ndarray, p_treated: float, trimming_threshold: float = 0.01
 656) -> pd.DataFrame:
 657    """
 658    Gateaux derivatives of the ATTE score wrt nuisances (g0, m). g1-derivative is 0.
 659
 660    For ψ_ATTE = [ D*(Y - g0 - θ)  -  (1-D)*(m/(1-m))*(Y - g0) ] / p_treated:
 661
 662      ∂_{g0}[h] : (1/n) Σ h(X_i) * [ ((1-D_i)*m_i/(1-m_i) - D_i) / p_treated ]
 663      ∂_{m}[s]  : (1/n) Σ s(X_i) * [ -(1-D_i)*(Y_i - g0_i) / ( p_treated * (1-m_i)^2 ) ]
 664
 665    Both have 0 expectation at the truth (Neyman orthogonality).
 666    """
 667    n, B = X_basis.shape
 668    m = np.clip(m, trimming_threshold, 1 - trimming_threshold)
 669    odds = m / (1.0 - m)
 670
 671    dg0_terms = X_basis * (((1.0 - d) * odds - d) / (p_treated + 1e-12))[:, None]
 672    dg0 = dg0_terms.mean(axis=0)
 673    dg0_se = dg0_terms.std(axis=0, ddof=1) / np.sqrt(n)
 674
 675    dm_terms = - X_basis * (((1.0 - d) * (y - g0)) / ((p_treated + 1e-12) * (1.0 - m)**2))[:, None]
 676    dm = dm_terms.mean(axis=0)
 677    dm_se = dm_terms.std(axis=0, ddof=1) / np.sqrt(n)
 678
 679    dg1 = np.zeros(B)
 680    dg1_se = np.zeros(B)
 681    t = lambda est, se: est / np.maximum(se, 1e-12)
 682
 683    return pd.DataFrame({
 684        "basis": np.arange(B),
 685        "d_g1": dg1, "se_g1": dg1_se, "t_g1": np.zeros(B),
 686        "d_g0": dg0, "se_g0": dg0_se, "t_g0": t(dg0, dg0_se),
 687        "d_m": dm,   "se_m": dm_se,   "t_m":  t(dm, dm_se),
 688    })

 693def overlap_diagnostics_atte(
 694    m: np.ndarray, d: np.ndarray, eps_list: List[float] = [0.95, 0.97, 0.98, 0.99]
 695) -> pd.DataFrame:
 696    """
 697    Key overlap metrics for ATTE: availability of suitable controls.
 698    Reports conditional shares: among CONTROLS, fraction with m(X) ≥ threshold; among TREATED, fraction with m(X) ≤ 1 - threshold.
 699    """
 700    rows = []
 701    m = np.asarray(m)
 702    d_bool = np.asarray(d).astype(bool)
 703    ctrl = ~d_bool
 704    trt = d_bool
 705    n_ctrl = int(ctrl.sum())
 706    n_trt = int(trt.sum())
 707    for thr in eps_list:
 708        pct_ctrl = float(100.0 * ((m[ctrl] >= thr).mean() if n_ctrl else np.nan))
 709        pct_trt = float(100.0 * ((m[trt] <= (1.0 - thr)).mean() if n_trt else np.nan))
 710        rows.append({
 711            "threshold": thr,
 712            "pct_controls_with_m_ge_thr": pct_ctrl,
 713            "pct_treated_with_m_le_1_minus_thr": pct_trt,
 714        })
 715    return pd.DataFrame(rows)

 720def trim_sensitivity_curve_atte(
 721    inference_fn: Callable[..., Dict[str, Any]],
 722    data: CausalData,
 723    m: np.ndarray, d: np.ndarray,
 724    thresholds: np.ndarray = np.linspace(0.90, 0.995, 12),
 725    **inference_kwargs
 726) -> pd.DataFrame:
 727    """
 728    Re-estimate θ while progressively trimming CONTROLS with large m(X).
 729    """
 730    df_full = data.get_df()
 731    rows = []
 732    for thr in thresholds:
 733        controls = (np.asarray(d).astype(bool) == False)
 734        high_p = (np.asarray(m) >= thr)
 735        drop = controls & high_p
 736        keep = ~drop
 737        df_trim = df_full.loc[keep].copy()
 738        data_trim = CausalData(
 739            df=df_trim, treatment=data.treatment.name,
 740            outcome=data.outcome.name, confounders=list(data.confounders) if data.confounders else None
 741        )
 742        res = inference_fn(data_trim, **inference_kwargs)
 743        rows.append({
 744            "trim_threshold": float(thr),
 745            "n": int(keep.sum()),
 746            "pct_dropped": float(100.0 * drop.mean()),
 747            "theta": float(res["coefficient"]),
 748            "se": float(res.get("std_error", np.nan))
 749        })
 750    return pd.DataFrame(rows)

 755def trim_sensitivity_curve_ate(
 756    m_hat: np.ndarray,
 757    D: np.ndarray,
 758    Y: np.ndarray,
 759    g0_hat: np.ndarray,
 760    g1_hat: np.ndarray,
 761    eps_grid: tuple[float, ...] = (0.0, 0.005, 0.01, 0.02, 0.05),
 762) -> pd.DataFrame:
 763    """
 764    Sensitivity of ATE estimate to propensity clipping epsilon (no re-fit).
 765
 766    For each epsilon in eps_grid, compute the AIPW/IRM ATE estimate using
 767    m_clipped = clip(m_hat, eps, 1-eps) over the full sample and report
 768    the plug-in standard error from the EIF.
 769
 770    Parameters
 771    ----------
 772    m_hat, D, Y, g0_hat, g1_hat : np.ndarray
 773        Cross-fitted nuisances and observed arrays.
 774    eps_grid : tuple[float, ...]
 775        Sequence of clipping thresholds ε to evaluate.
 776
 777    Returns
 778    -------
 779    pd.DataFrame
 780        Columns: ['trim_eps','n','pct_clipped','theta','se'].
 781        pct_clipped is the percent of observations with m outside [ε,1-ε].
 782    """
 783    m = np.asarray(m_hat, dtype=float).ravel()
 784    d = np.asarray(D, dtype=float).ravel()
 785    y = np.asarray(Y, dtype=float).ravel()
 786    g0 = np.asarray(g0_hat, dtype=float).ravel()
 787    g1 = np.asarray(g1_hat, dtype=float).ravel()
 788    n = y.size
 789    rows: list[dict[str, float]] = []
 790
 791    for eps in eps_grid:
 792        eps_f = float(eps)
 793        m_clip = np.clip(m, eps_f, 1.0 - eps_f)
 794        # share clipped (either side)
 795        pct_clipped = float(100.0 * np.mean((m <= eps_f) | (m >= 1.0 - eps_f))) if n > 0 else float("nan")
 796        # AIPW ATE with clipped propensity
 797        theta_terms = (g1 - g0) + d * (y - g1) / m_clip - (1.0 - d) * (y - g0) / (1.0 - m_clip)
 798        theta = float(np.mean(theta_terms)) if n > 0 else float("nan")
 799        psi = theta_terms - theta
 800        se = float(np.std(psi, ddof=1) / np.sqrt(n)) if n > 1 else 0.0
 801        rows.append({
 802            "trim_eps": eps_f,
 803            "n": int(n),
 804            "pct_clipped": pct_clipped,
 805            "theta": theta,
 806            "se": se,
 807        })
 808
 809    return pd.DataFrame(rows)

 816def _is_binary(series: pd.Series) -> bool:
 817    """
 818    Determine if a pandas Series contains binary data (0/1 or True/False).
 819    
 820    Parameters
 821    ----------
 822    series : pd.Series
 823        The series to check
 824        
 825    Returns
 826    -------
 827    bool
 828        True if the series appears to be binary
 829    """
 830    unique_vals = set(series.dropna().unique())
 831    
 832    # Check for 0/1 binary
 833    if unique_vals == {0, 1} or unique_vals == {0} or unique_vals == {1}:
 834        return True
 835    
 836    # Check for True/False binary
 837    if unique_vals == {True, False} or unique_vals == {True} or unique_vals == {False}:
 838        return True
 839        
 840    # Check for numeric that could be treated as binary (only two distinct values)
 841    if len(unique_vals) == 2 and all(isinstance(x, (int, float, bool, np.integer, np.floating)) for x in unique_vals):
 842        return True
 843        
 844    return False

 847def _generate_random_outcome(original_outcome: pd.Series, rng: np.random.Generator) -> np.ndarray:
 848    """
 849    Generate random outcome variables matching the distribution of the original outcome.
 850    
 851    For binary outcomes: generates random binary variables with same proportion as original
 852    For continuous outcomes: generates random continuous variables from normal distribution
 853    fitted to the original data
 854    
 855    Parameters
 856    ----------
 857    original_outcome : pd.Series
 858        The original outcome variable
 859    rng : np.random.Generator
 860        Random number generator
 861        
 862    Returns
 863    -------
 864    np.ndarray
 865        Generated random outcome variables
 866    """
 867    n = len(original_outcome)
 868    
 869    if _is_binary(original_outcome):
 870        # For binary outcome, generate random binary with same proportion
 871        original_rate = float(original_outcome.mean())
 872        return rng.binomial(1, original_rate, size=n).astype(original_outcome.dtype)
 873    else:
 874        # For continuous outcome, generate from normal distribution fitted to original data
 875        mean = float(original_outcome.mean())
 876        std = float(original_outcome.std())
 877        if std == 0:
 878            # If no variance, generate constant values equal to the mean
 879            return np.full(n, mean, dtype=original_outcome.dtype)
 880        return rng.normal(mean, std, size=n).astype(original_outcome.dtype)

 883def _generate_random_treatment(original_treatment: pd.Series, rng: np.random.Generator) -> np.ndarray:
 884    """
 885    Generate random binary treatment variables with same proportion as original.
 886    
 887    Parameters
 888    ----------
 889    original_treatment : pd.Series
 890        The original treatment variable
 891    rng : np.random.Generator

 897        Generated random binary treatment variables
 898    """

 900    treatment_rate = float(original_treatment.mean())

 904def _run_inference(
 905    inference_fn: Callable[..., Dict[str, Any]],
 906    data: CausalData,
 907    **kwargs,
 908) -> Dict[str, float]:
 909    """
 910    Helper that executes `inference_fn` and extracts the two items of interest.
 911    Provides a safe fallback for p_value if missing by using a normal approximation
 912    based on coefficient and std_error.
 913    """
 914    res = inference_fn(data, **kwargs)
 915    out = {"theta": float(res["coefficient"])}
 916    p = res.get("p_value")
 917    if p is None and "std_error" in res:
 918        try:
 919            se = float(res["std_error"])
 920            th = float(res["coefficient"])
 921            z = 0.0 if se == 0.0 else th / se
 922            p = 2.0 * (1 - stats.norm.cdf(abs(z)))
 923        except Exception:
 924            p = None
 925    out["p_value"] = float(p) if p is not None else float("nan")
 926    return out

1034def _fast_fold_thetas_from_psi(psi_a: np.ndarray, psi_b: np.ndarray, fold_indices: List[np.ndarray]) -> List[float]:
1035    """
1036    Compute leave-fold-out θ_{-k} using cached ψ_a, ψ_b.
1037    θ_{-k} = - mean_R(ψ_b) / mean_R(ψ_a), where R is the complement of fold k.
1038    """
1039    psi_a = np.asarray(psi_a, dtype=float).ravel()
1040    psi_b = np.asarray(psi_b, dtype=float).ravel()
1041    n = psi_a.size
1042    sum_a = float(psi_a.sum())
1043    sum_b = float(psi_b.sum())
1044    thetas: List[float] = []
1045    for idx in fold_indices:
1046        idx = np.asarray(idx, dtype=int)
1047        n_r = n - idx.size
1048        if n_r <= 0:
1049            thetas.append(0.0)
1050            continue
1051        sa = sum_a - float(psi_a[idx].sum())
1052        sb = sum_b - float(psi_b[idx].sum())
1053        mean_a = sa / n_r
1054        mean_b = sb / n_r
1055        denom = mean_a if abs(mean_a) > 1e-12 else -1.0  # robust guard against div-by-zero
1056        thetas.append(-mean_b / denom)
1057    return thetas

1060def oos_moment_check_from_psi(
1061    psi_a: np.ndarray,
1062    psi_b: np.ndarray,
1063    fold_indices: List[np.ndarray],
1064    *,
1065    strict: bool = False,
1066) -> Tuple[pd.DataFrame, float, Optional[float]]:
1067    """
1068    OOS moment check using cached ψ_a, ψ_b only.
1069    Returns (fold-wise DF, t_fold_agg, t_strict if requested).
1070    """
1071    psi_a = np.asarray(psi_a, dtype=float).ravel()
1072    psi_b = np.asarray(psi_b, dtype=float).ravel()
1073    fold_thetas = _fast_fold_thetas_from_psi(psi_a, psi_b, fold_indices)
1074
1075    rows: List[Dict[str, Any]] = []
1076    total_n = 0
1077    total_sum = 0.0
1078    total_sumsq = 0.0
1079
1080    for k, idx in enumerate(fold_indices):
1081        idx = np.asarray(idx, dtype=int)
1082        th = fold_thetas[k]
1083        # ψ_k = ψ_b[idx] + ψ_a[idx]*θ_{-k}
1084        psi_k = psi_b[idx] + psi_a[idx] * th
1085        n_k = psi_k.size
1086        m_k = float(psi_k.mean()) if n_k > 0 else 0.0
1087        v_k = float(psi_k.var(ddof=1)) if n_k > 1 else 0.0
1088        rows.append({"fold": k, "n": n_k, "psi_mean": m_k, "psi_var": v_k})
1089
1090        if strict and n_k > 0:
1091            total_n += n_k
1092            s = float(psi_k.sum())
1093            total_sum += s
1094            total_sumsq += float((psi_k * psi_k).sum())
1095
1096    df = pd.DataFrame(rows)
1097
1098    # Fold-aggregated t-stat
1099    num = float((df["n"] * df["psi_mean"]).sum())
1100    den = float(np.sqrt((df["n"] * df["psi_var"]).sum()))
1101    t_fold = (num / den) if den > 0 else 0.0
1102
1103    t_strict: Optional[float] = None
1104    if strict and total_n > 1:
1105        mean_all = total_sum / total_n
1106        var_all = (total_sumsq - total_n * (mean_all ** 2)) / (total_n - 1)
1107        se_all = float(np.sqrt(max(var_all, 0.0) / total_n))
1108        t_strict = (mean_all / se_all) if se_all > 0 else 0.0
1109
1110    return df, float(t_fold), (None if t_strict is None else float(t_strict))

1235) -> Dict[str, Any]:

1375def _grade(val: float, warn: float, strong: float, *, larger_is_worse: bool = True) -> str:
1376    """Map a scalar to GREEN/YELLOW/RED with optional direction."""
1377    if val is None or (isinstance(val, float) and (np.isnan(val) or np.isinf(val))):
1378        return "NA"
1379    v = float(val)
1380    if larger_is_worse:
1381        return "GREEN" if v < warn else ("YELLOW" if v < strong else "RED")
1382    else:
1383        # smaller is worse → invert thresholds
1384        return "GREEN" if v <= warn else ("YELLOW" if v <= strong else "RED")

1387def add_score_flags(rep_score: dict, thresholds: dict | None = None, *, effect_size_guard: float = 0.02, oos_gate: bool = True, se_rule: str | None = None, se_ref: float | None = None) -> dict:

1392      - rep['summary'] with a new 'flag' column
1393      - rep['overall_flag'] (rollup)
1394
1395    Additional logic:
1396      - Practical effect-size guard: if the constant-basis derivative magnitude is tiny
1397        (<= effect_size_guard), then downgrade an orthogonality RED to GREEN (if OOS is GREEN)
1398        or to YELLOW (otherwise). Controlled by `oos_gate`.

1404    # --- defaults tuned to your style (>= means worse) ---
1405    thr = {
1406        # heavy tails / stability
1407        "tail_ratio_warn": 10.0,   # |psi| p99 / median
1408        "tail_ratio_strong": 20.0,
1409        "kurt_warn": 10.0,         # normal≈3; allow slack for EIFs

1413        "t_warn": 2.0,
1414        "t_strong": 4.0,
1415
1416        # OOS moment t-stat (|t| should be ~0)
1417        "oos_warn": 2.0,
1418        "oos_strong": 3.0,
1419    }
1420    if thresholds:
1421        thr.update(thresholds)
1422
1423    # --- fetch raw pieces robustly ---
1424    infl = rep.get("influence_diagnostics", {}) or {}
1425    ortho = rep.get("orthogonality_derivatives")
1426    oos   = rep.get("oos_moment_test")
1427
1428    # Influence metrics
1429    tail_ratio = infl.get("p99_over_med") if isinstance(infl, dict) else None
1430    kurtosis   = infl.get("kurtosis") if isinstance(infl, dict) else None
1431
1432    # Max |t| over bases (compute if not already summarized)
1433    def _max_abs(df: pd.DataFrame, col: str) -> float | float:
1434        if isinstance(df, pd.DataFrame) and col in df.columns:
1435            a = np.abs(df[col].to_numpy(dtype=float))
1436            return float(np.nanmax(a)) if a.size else float("nan")
1437        return float("nan")

1439    max_t_g1 = _max_abs(ortho, "t_g1")
1440    max_t_g0 = _max_abs(ortho, "t_g0")
1441    max_t_m  = _max_abs(ortho, "t_m")
1442
1443    # OOS t-stats (if available)
1444    oos_t_fold   = float(oos.get("tstat_fold_agg")) if isinstance(oos, dict) and "tstat_fold_agg" in oos else float("nan")
1445    oos_t_strict = float(oos.get("tstat_strict"))   if isinstance(oos, dict) and "tstat_strict"   in oos and oos.get("tstat_strict") is not None else float("nan")
1446
1447    # --- compute flags ---
1448    flags = {
1449        # influence / tails
1450        "psi_tail_ratio": _grade(tail_ratio, thr["tail_ratio_warn"], thr["tail_ratio_strong"], larger_is_worse=True),
1451        "psi_kurtosis":   _grade(kurtosis,   thr["kurt_warn"],       thr["kurt_strong"],       larger_is_worse=True),
1452

1454        "ortho_max_|t|_g1": _grade(max_t_g1, thr["t_warn"], thr["t_strong"], larger_is_worse=True),
1455        "ortho_max_|t|_g0": _grade(max_t_g0, thr["t_warn"], thr["t_strong"], larger_is_worse=True),
1456        "ortho_max_|t|_m":  _grade(max_t_m,  thr["t_warn"], thr["t_strong"], larger_is_worse=True),

655    X_basis: np.ndarray, y: np.ndarray, d: np.ndarray,
656    g0: np.ndarray, m: np.ndarray, p_treated: float, trimming_threshold: float = 0.01
657) -> pd.DataFrame:
658    """
659    Gateaux derivatives of the ATTE score wrt nuisances (g0, m). g1-derivative is 0.
660
661    For ψ_ATTE = [ D*(Y - g0 - θ)  -  (1-D)*(m/(1-m))*(Y - g0) ] / p_treated:
662
663      ∂_{g0}[h] : (1/n) Σ h(X_i) * [ ((1-D_i)*m_i/(1-m_i) - D_i) / p_treated ]
664      ∂_{m}[s]  : (1/n) Σ s(X_i) * [ -(1-D_i)*(Y_i - g0_i) / ( p_treated * (1-m_i)^2 ) ]
665
666    Both have 0 expectation at the truth (Neyman orthogonality).

674    dg0_se = dg0_terms.std(axis=0, ddof=1) / np.sqrt(n)
675

678    dm_se = dm_terms.std(axis=0, ddof=1) / np.sqrt(n)
679

681    dg1_se = np.zeros(B)
682    t = lambda est, se: est / np.maximum(se, 1e-12)
683
684    return pd.DataFrame({

686        "d_g1": dg1, "se_g1": dg1_se, "t_g1": np.zeros(B),
687        "d_g0": dg0, "se_g0": dg0_se, "t_g0": t(dg0, dg0_se),

689    })
