497    def _center(a: np.ndarray) -> np.ndarray:
498        return a - np.mean(a)

500    def _center_w(a: np.ndarray, w: np.ndarray) -> np.ndarray:

507    def _ols_r2_and_fit(yv: np.ndarray, Z: np.ndarray, w: Optional[np.ndarray] = None) -> tuple[float, np.ndarray]:
508        """Stable (weighted) OLS on centered & standardized vars for R^2 and fitted component."""
509        Z = np.asarray(Z, dtype=float)
510        if Z.ndim == 1:
511            Z = Z.reshape(-1, 1)
512        if w is None:
513            # Unweighted
514            yv_c = _center(yv)
515            Zc = Z - np.nanmean(Z, axis=0, keepdims=True)
516            col_std = np.nanstd(Zc, axis=0, ddof=0)
517            valid = np.isfinite(col_std) & (col_std > 1e-12)
518            if not np.any(valid):
519                return 0.0, np.zeros_like(yv_c)
520            Zcs = Zc[:, valid] / col_std[valid]
521            Zcs = np.nan_to_num(Zcs, nan=0.0, posinf=0.0, neginf=0.0)
522            yv_c = np.nan_to_num(np.asarray(yv_c, float), nan=0.0, posinf=0.0, neginf=0.0)
523            from numpy.linalg import lstsq
524            beta, *_ = lstsq(Zcs, yv_c, rcond=1e-12)
525            yhat = Zcs @ beta
526            denom = float(np.dot(yv_c, yv_c))
527            if not np.isfinite(denom) or denom <= 1e-12:
528                return 0.0, np.zeros_like(yv_c)
529            num = float(np.dot(yhat, yhat))
530            if not np.isfinite(num) or num < 0.0:
531                return 0.0, np.zeros_like(yv_c)
532            r2 = float(np.clip(num / denom, 0.0, 1.0))
533            return r2, yhat
534        else:
535            # Weighted
536            w = np.asarray(w, float)
537            # sanitize weights and features to avoid NaN/inf propagation
538            w = np.nan_to_num(w, nan=0.0, posinf=0.0, neginf=0.0)
539            Z = np.asarray(Z, float)
540            Z = np.nan_to_num(Z, nan=0.0, posinf=0.0, neginf=0.0)
541            sw = float(np.sum(w))
542            if not np.isfinite(sw) or sw <= 1e-12:
543                return 0.0, np.zeros_like(yv, dtype=float)
544            yv_c = _center_w(yv, w)
545            # Weighted column means
546            muZ = (w[:, None] * Z).sum(axis=0) / sw
547            Zc = Z - muZ
548            # Weighted std per column
549            var = (w[:, None] * (Zc * Zc)).sum(axis=0) / sw
550            std = np.sqrt(np.maximum(var, 0.0))
551            valid = np.isfinite(std) & (std > 1e-12)
552            if not np.any(valid):
553                return 0.0, np.zeros_like(yv_c)
554            Zcs = Zc[:, valid] / std[valid]
555            Zcs = np.nan_to_num(Zcs, nan=0.0, posinf=0.0, neginf=0.0)
556            yv_c = np.nan_to_num(np.asarray(yv_c, float), nan=0.0, posinf=0.0, neginf=0.0)
557            # Weighted least squares via sqrt(w)
558            swr = np.sqrt(np.clip(w, 0.0, np.inf))
559            Zsw = Zcs * swr[:, None]
560            ysw = yv_c * swr
561            from numpy.linalg import lstsq
562            beta, *_ = lstsq(Zsw, ysw, rcond=1e-12)
563            yhat = Zcs @ beta
564            denom = float(np.dot(ysw, ysw))
565            if not np.isfinite(denom) or denom <= 1e-12:
566                return 0.0, np.zeros_like(yv_c)
567            num = float(np.dot(swr * yhat, swr * yhat))
568            if not np.isfinite(num) or num < 0.0:
569                return 0.0, np.zeros_like(yv_c)
570            r2 = float(np.clip(num / denom, 0.0, 1.0))
571            return r2, yhat

585    def _safe_corr(u: np.ndarray, v: np.ndarray, w: Optional[np.ndarray] = None) -> float:
586        if w is None:
587            u = _center(u); v = _center(v)
588            su, sv = np.std(u), np.std(v)
589            if not (np.isfinite(su) and np.isfinite(sv)) or su <= 0 or sv <= 0:
590                return 0.0
591            val = float(np.corrcoef(u, v)[0, 1])
592            return float(np.clip(val, -1.0, 1.0))
593        u = _center_w(u, w); v = _center_w(v, w)
594        sw = float(np.sum(w))
595        su = np.sqrt(max(0.0, float(np.sum(w * u * u)) / (sw if sw > 1e-12 else 1.0)))
596        sv = np.sqrt(max(0.0, float(np.sum(w * v * v)) / (sw if sw > 1e-12 else 1.0)))
597        if su <= 0 or sv <= 0:
598            return 0.0
599        cov = float(np.sum(w * u * v)) / (sw if sw > 1e-12 else 1.0)
600        val = cov / (su * sv)
601        return float(np.clip(val, -1.0, 1.0))

Computes a benchmark for a given set of features by refitting a short IRM model (excluding the provided features) and contrasting it with the original (long) model. Returns a DataFrame containing cf_y, cf_d, rho and the change in estimates.

Parameters
  • effect_estimation (dict): A dictionary containing the fitted IRM model under the key 'model'.
  • benchmarking_set (list[str]): List of confounder names to be used for benchmarking (to be removed in the short model).
  • fit_args (dict, optional): Additional keyword arguments for the IRM.fit() method of the short model.
Returns
  • pandas.DataFrame: A one-row DataFrame indexed by the treatment name with columns:
    • cf_y, cf_d, rho: residual-based benchmarking strengths
    • theta_long, theta_short, delta: effect estimates and their change (long - short)
