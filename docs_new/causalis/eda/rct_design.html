 19    """
 20    Calculate the Minimum Detectable Effect (MDE) for conversion or continuous data.
 21    
 22    Parameters
 23    ----------
 24    sample_size : int or tuple of int
 25        Total sample size or a tuple of (control_size, treatment_size).
 26        If a single integer is provided, the sample will be split according to the ratio parameter.
 27    baseline_rate : float, optional
 28        Baseline conversion rate (for conversion data) or baseline mean (for continuous data).
 29        Required for conversion data.
 30    variance : float or tuple of float, optional
 31        Variance of the data. For conversion data, this is calculated from the baseline rate if not provided.
 32        For continuous data, this parameter is required.
 33        Can be a single float (assumed same for both groups) or a tuple of (control_variance, treatment_variance).
 34    alpha : float, default 0.05
 35        Significance level (Type I error rate).
 36    power : float, default 0.8
 37        Statistical power (1 - Type II error rate).
 38    data_type : str, default 'conversion'
 39        Type of data. Either 'conversion' for binary/conversion data or 'continuous' for continuous data.
 40    ratio : float, default 0.5
 41        Ratio of the sample allocated to the control group if sample_size is a single integer.
 42        
 43    Returns
 44    -------
 45    Dict[str, Any]
 46        A dictionary containing:
 47        - 'mde': The minimum detectable effect (absolute)
 48        - 'mde_relative': The minimum detectable effect as a percentage of the baseline (relative)
 49        - 'parameters': The parameters used for the calculation
 50        
 51    Examples
 52    --------
 53    >>> # Calculate MDE for conversion data with 1000 total sample size and 10% baseline conversion rate
 54    >>> calculate_mde(1000, baseline_rate=0.1, data_type='conversion')
 55    {'mde': 0.0527..., 'mde_relative': 0.5272..., 'parameters': {...}}
 56    
 57    >>> # Calculate MDE for continuous data with 500 samples in each group and variance of 4
 58    >>> calculate_mde((500, 500), variance=4, data_type='continuous')
 59    {'mde': 0.3482..., 'mde_relative': None, 'parameters': {...}}
 60    
 61    Notes
 62    -----
 63    For conversion data, the MDE is calculated using the formula:
 64    MDE = (z_α/2 + z_β) * sqrt((p1*(1-p1)/n1) + (p2*(1-p2)/n2))
 65    
 66    For continuous data, the MDE is calculated using the formula:
 67    MDE = (z_α/2 + z_β) * sqrt((σ1²/n1) + (σ2²/n2))
 68    
 69    where:
 70    - z_α/2 is the critical value for significance level α
 71    - z_β is the critical value for power
 72    - p1 and p2 are the conversion rates in the control and treatment groups
 73    - σ1² and σ2² are the variances in the control and treatment groups
 74    - n1 and n2 are the sample sizes in the control and treatment groups
 75    """
 76    # Validate inputs
 77    if data_type not in ['conversion', 'continuous']:
 78        raise ValueError("data_type must be either 'conversion' or 'continuous'")
 79    
 80    if data_type == 'conversion' and baseline_rate is None:
 81        raise ValueError("baseline_rate is required for conversion data")
 82    
 83    if data_type == 'continuous' and variance is None:
 84        raise ValueError("variance is required for continuous data")
 85    
 86    # Calculate critical values
 87    z_alpha = abs(np.percentile(np.random.normal(0, 1, 100000), (1 - alpha / 2) * 100))
 88    z_beta = abs(np.percentile(np.random.normal(0, 1, 100000), power * 100))
 89    
 90    # Determine sample sizes for control and treatment groups
 91    if isinstance(sample_size, tuple):
 92        n_control, n_treatment = sample_size
 93    else:
 94        n_control = int(sample_size * ratio)
 95        n_treatment = sample_size - n_control
 96    
 97    # Calculate MDE based on data type
 98    if data_type == 'conversion':
 99        # For conversion data
100        p_control = baseline_rate
101        
102        # If variance is not provided, calculate it from the baseline rate
103        if variance is None:
104            var_control = p_control * (1 - p_control)
105            var_treatment = p_control * (1 - p_control)  # Assuming same variance for treatment
106        elif isinstance(variance, tuple):
107            var_control, var_treatment = variance
108        else:
109            var_control = var_treatment = variance
110        
111        # Calculate MDE
112        mde = (z_alpha + z_beta) * np.sqrt((var_control / n_control) + (var_treatment / n_treatment))
113        
114        # Calculate relative MDE
115        mde_relative = mde / baseline_rate if baseline_rate > 0 else None
116        
117    else:  # data_type == 'continuous'
118        # For continuous data
119        if isinstance(variance, tuple):
120            var_control, var_treatment = variance
121        else:
122            var_control = var_treatment = variance
123        
124        # Calculate MDE
125        mde = (z_alpha + z_beta) * np.sqrt((var_control / n_control) + (var_treatment / n_treatment))
126        
127        # Calculate relative MDE if baseline_rate is provided
128        mde_relative = mde / baseline_rate if baseline_rate is not None and baseline_rate != 0 else None
129    
130    # Prepare the result dictionary
131    result = {
132        'mde': mde,
133        'mde_relative': mde_relative,
134        'parameters': {
135            'sample_size': {
136                'total': n_control + n_treatment,
137                'control': n_control,
138                'treatment': n_treatment
139            },
140            'alpha': alpha,
141            'power': power,
142            'data_type': data_type
143        }
144    }
145    
146    # Add data-type specific parameters
147    if data_type == 'conversion':
148        result['parameters']['baseline_rate'] = baseline_rate
149    else:  # data_type == 'continuous'
150        result['parameters']['variance'] = {
151            'control': var_control,
152            'treatment': var_treatment
153        }
154        if baseline_rate is not None:
155            result['parameters']['baseline_mean'] = baseline_rate
156    
157    return result

 64    """
 65    Deterministically assign variants for each row in df based on id_col.
 66
 67    Parameters
 68    ----------
 69    df : pd.DataFrame
 70        Input DataFrame with an identifier column.
 71    id_col : str
 72        Column name in df containing entity identifiers (user_id, session_id, etc.).
 73    experiment_id : str
 74        Unique identifier for the experiment (versioned for reruns).
 75    variants : Dict[str, float]
 76        Mapping from variant name to weight (coverage). Weights must be non-negative
 77        and their sum must be in (0, 1]. If the sum is < 1, the remaining mass
 78        corresponds to "not in experiment" and the assignment will be None.
 79    salt : str, default "global_ab_salt"
 80        Secret string to de-correlate from other hash uses and make assignments
 81        non-gameable.
 82    layer_id : str, default "default"
 83        Identifier for mutual exclusivity layer or surface. In this case work like
 84        another random
 85    variant_col : str, default "variant"
 86        Name of output column to store assigned variant labels.
 87
 88    Returns
 89    -------
 90    pd.DataFrame
 91        A copy of df with an extra column `variant_col`.
 92        Entities outside experiment coverage will have None in the variant column.
 93    """
 94    if id_col not in df.columns:
 95        raise ValueError(f"id_col '{id_col}' not found in DataFrame")
 96
 97    _validate_variants(variants)
 98
 99    def _row_assign(entity_id: str) -> Optional[str]:
100        key = f"{salt}|{layer_id}|{experiment_id}|{entity_id}"
101        u = _hash_to_unit_interval(key)
102
103        cumulative = 0.0
104        for name, weight in sorted(variants.items()):
105            if weight == 0:
106                continue
107            cumulative += float(weight)
108            if u < cumulative:
109                return name
110
111        return None  # Not in experiment
112
113    out = df.copy()
114    out[variant_col] = df[id_col].astype(str).map(_row_assign)
115    return out

def check_srm( assignments: Iterable[Hashable] | pandas.core.series.Series, target_allocation: Dict[Hashable, int | float], alpha: float = 0.001, min_expected: float = 5.0, strict_variants: bool = True) -> SRMResult: 
 50def check_srm(
 51    assignments: Union[Iterable[Hashable], pd.Series],
 52    target_allocation: Dict[Hashable, Number],
 53    alpha: float = 1e-3,
 54    min_expected: float = 5.0,
 55    strict_variants: bool = True,
 56) -> SRMResult:
 57    """
 58    Check Sample Ratio Mismatch (SRM) for an RCT via chi-square goodness-of-fit test.
 59
 60    Parameters
 61    ----------
 62    assignments:
 63        Iterable of assigned variant labels for each unit (user_id, session_id, etc.).
 64        E.g. Series of ["control", "treatment", ...].
 65
 66    target_allocation:
 67        Mapping {variant: p} describing intended allocation as PROBABILITIES.
 68        - Each p must be > 0.
 69        - Sum of all p must be 1.0 (within numerical tolerance).
 70
 71        Examples:
 72            {"control": 0.5, "treatment": 0.5}
 73            {"A": 0.2, "B": 0.3, "C": 0.5}
 74
 75    alpha:
 76        Significance level. Use strict values like 1e-3 or 1e-4 in production.
 77
 78    min_expected:
 79        If any expected count < min_expected, a warning is attached.
 80
 81    strict_variants:
 82        - True: fail if observed variants differ from target keys.
 83        - False: drop unknown variants and test only on declared ones.
 84
 85    Returns
 86    -------
 87    SRMResult
 88    """
 89    # --- Prepare data
 90    s = pd.Series(list(assignments)).dropna()
 91    if s.empty:
 92        raise ValueError("No assignments provided for SRM check.")
 93
 94    if not target_allocation:
 95        raise ValueError("target_allocation cannot be empty.")
 96
 97    # Validate probabilities
 98    probs = np.array(list(target_allocation.values()), dtype=float)
 99
100    if (probs <= 0).any():
101        raise ValueError("All target allocation probabilities must be > 0.")
102
103    total = float(probs.sum())
104    if not np.isclose(total, 1.0, rtol=1e-6, atol=1e-8):
105        raise ValueError(
106            f"target_allocation probabilities must sum to 1.0, got {total:.6f}."
107        )
108
109    variants = list(target_allocation.keys())
110    target_map = dict(zip(variants, probs))
111
112    # Observed counts
113    if strict_variants:
114        unexpected = set(s.unique()) - set(variants)
115        if unexpected:
116            raise ValueError(
117                f"Found assignments to variants not in target_allocation: {unexpected}"
118            )
119
120    if not strict_variants:
121        s = s[s.isin(variants)]
122        if s.empty:
123            raise ValueError(
124                "After filtering to target variants, no assignments remain."
125            )
126
127    observed_counts = s.value_counts().reindex(variants).fillna(0).astype(int)
128    n = int(observed_counts.sum())
129    if n == 0:
130        raise ValueError("Total sample size is zero after preprocessing.")
131
132    # Expected counts from probabilities * n
133    expected_counts = np.array(
134        [target_map[v] * n for v in variants],
135        dtype=float
136    )
137
138    # Chi-square statistic
139    with np.errstate(divide="ignore", invalid="ignore"):
140        chi2_components = (observed_counts.values - expected_counts) ** 2 / expected_counts
141    chi2_components = np.nan_to_num(chi2_components, nan=0.0, posinf=0.0)
142    chi2_stat = float(chi2_components.sum())
143
144    # Degrees of freedom
145    k = (expected_counts > 0).sum()
146    df = max(int(k) - 1, 1)
147
148    if chi2 is None:
149        raise ImportError(
150            "scipy is required for p-value computation in check_srm(). "
151            f"Original error: {_scipy_import_error}"
152        )
153
154    p_value = float(chi2.sf(chi2_stat, df))
155
156    warning = None
157    if (expected_counts < min_expected).any():
158        warning = (
159            f"Some expected cell counts are < {min_expected:.1f}. "
160            "Chi-square approximation may be unreliable; "
161            "consider exact or simulation-based tests."
162        )
163
164    is_srm = p_value < alpha
165
166    return SRMResult(
167        chi2=chi2_stat,
168        df=df,
169        p_value=p_value,
170        expected={v: float(e) for v, e in zip(variants, expected_counts)},
171        observed={v: int(o) for v, o in zip(variants, observed_counts)},
172        alpha=alpha,
173        is_srm=is_srm,
174        warning=warning,
175    )
