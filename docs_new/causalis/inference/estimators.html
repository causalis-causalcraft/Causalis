226    def _get_weights(self, n: int, m_hat_adj: Optional[np.ndarray], d: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
227        # Standard ATE
228        if self.score == "ATE":
229            if self.weights is None:
230                w = np.ones(n, dtype=float)
231            elif isinstance(self.weights, np.ndarray):
232                if self.weights.shape[0] != n:
233                    raise ValueError("weights array must have shape (n,)")
234                w = np.asarray(self.weights, dtype=float)
235            elif isinstance(self.weights, dict):
236                w = np.asarray(self.weights.get("weights"), dtype=float)
237                if w.shape[0] != n:
238                    raise ValueError("weights['weights'] must have shape (n,)")
239            else:
240                raise TypeError("weights must be None, np.ndarray, or dict")
241            w_bar = w
242            if isinstance(self.weights, dict) and "weights_bar" in self.weights:
243                w_bar = np.asarray(self.weights["weights_bar"], dtype=float)
244                if w_bar.ndim == 2:
245                    # choose first repetition for now
246                    w_bar = w_bar[:, 0]
247            return w, w_bar
248        # ATTE requires m_hat
249        elif self.score == "ATTE":
250            if m_hat_adj is None:
251                raise ValueError("m_hat required for ATTE weights computation")
252            base_w = np.ones(n, dtype=float)
253            subgroup = base_w * d
254            subgroup_prob = float(np.mean(subgroup)) if np.mean(subgroup) > 0 else 1.0
255            w = subgroup / subgroup_prob
256            w_bar = (m_hat_adj * base_w) / subgroup_prob
257            return w, w_bar

261    def _normalize_ipw_terms(self, d: np.ndarray, m_hat: np.ndarray) -> Tuple[np.ndarray, np.ndarray]:
262        # Compute IPW terms and optionally normalize to mean 1
263        h1 = d / m_hat
264        h0 = (1 - d) / (1 - m_hat)
265        if self.normalize_ipw:
266            h1_mean = np.mean(h1)
267            h0_mean = np.mean(h0)
268            # Avoid division by zero
269            h1 = h1 / (h1_mean if h1_mean != 0 else 1.0)
270            h0 = h0 / (h0_mean if h0_mean != 0 else 1.0)
271        return h1, h0

493    def _sensitivity_element_est(self) -> dict:
494        """Compute elements needed for sensitivity bias bounds.
495        Mirrors DoubleMLIRM._sensitivity_element_est using fitted nuisances.
496        Requires fit() to have been called.
497        """
498        if any(getattr(self, attr) is None for attr in ["g0_hat_", "g1_hat_", "m_hat_", "coef_"]):
499            raise RuntimeError("IRM model must be fitted before sensitivity analysis.")
500        y = self._y
501        d = self._d
502        if y is None or d is None:
503            # fallback to current data
504            df = self.data.get_df()
505            y = df[self.data.outcome.name].to_numpy(dtype=float)
506            d = df[self.data.treatment.name].to_numpy(dtype=int)
507        m_hat = np.asarray(self.m_hat_, dtype=float)
508        g0 = np.asarray(self.g0_hat_, dtype=float)
509        g1 = np.asarray(self.g1_hat_, dtype=float)
510
511        # Adjusted propensity (we use clipped m_hat)
512        m_hat_adj = m_hat
513        # weights
514        w, w_bar = self._get_weights(n=len(y), m_hat_adj=m_hat_adj, d=d)
515
516        # sigma^2
517        sigma2_score_element = np.square(y - d * g1 - (1.0 - d) * g0)
518        sigma2 = float(np.mean(sigma2_score_element))
519        psi_sigma2 = sigma2_score_element - sigma2
520
521        # Riesz representer and m_alpha
522        with np.errstate(divide='ignore', invalid='ignore'):
523            inv_m = 1.0 / m_hat_adj
524            inv_1m = 1.0 / (1.0 - m_hat_adj)
525        m_alpha = w * w_bar * (inv_m + inv_1m)
526        rr = w_bar * (d * inv_m - (1.0 - d) * inv_1m)
527
528        nu2_score_element = 2.0 * m_alpha - np.square(rr)
529        nu2 = float(np.mean(nu2_score_element))
530        psi_nu2 = nu2_score_element - nu2
531
532        return {
533            "sigma2": sigma2,
534            "nu2": nu2,
535            "psi_sigma2": psi_sigma2,
536            "psi_nu2": psi_nu2,
537            "riesz_rep": rr,
538            "m_alpha": m_alpha,
539        }

576        z = norm.ppf(0.5 + level / 2.0)
577        ci_low = theta_hat - z * se
578        ci_high = theta_hat + z * se
579

581        theta_upper = theta_hat + max_bias
582

596        lines.append("")
597        lines.append("------------------ Robustness (SNR proxy) -------------")

Parameters

 12    r"""Best linear predictor (BLP) with orthogonal signals.
 13    Mainly used for CATE and GATE estimation for IRM models.
 14
 15    The Best Linear Predictor (BLP) targets the coefficient vector :math:`\beta_0` that minimizes the mean squared error
 16    between the true treatment effect function :math:`\tau(X)` and a linear combination of basis functions :math:`b(X)`:
 17
 18    .. math::
 19        \beta_0 = \arg\min_{\beta \in \mathbb{R}^K} \mathbb{E}\Big[\big(\tau(X) - b(X)^\top \beta \big)^2\Big].
 20
 21    This is characterized by the moment condition:
 22
 23    .. math::
 24        \mathbb{E}[b(X)\psi] = \mathbb{E}[b(X)b(X)^\top]\beta_0,
 25
 26    where :math:`\psi` is the orthogonal signal such that :math:`\mathbb{E}[\psi \mid X] = \tau(X)`.
 27
 28    The estimator is obtained via OLS of the orthogonal signal on the basis:
 29
 30    .. math::
 31        \hat{\beta} = (B^\top B)^{-1}B^\top\psi.
 32
 33    **GATE (Group Average Treatment Effect)**
 34
 35    When ``is_gate=True``, the basis consists of group indicators (dummy variables).
 36    In this case, the BLP coefficients correspond to the group means of the orthogonal signal,
 37    which approximate the GATEs:
 38
 39    .. math::
 40        \hat{\beta}_k = \frac{1}{n_k}\sum_{i:G_i=k}\psi_i \approx \text{GATE}_k.
 41
 42    **Confidence Intervals**
 43
 44    Confidence intervals for any linear combination :math:`\hat{g} = A\hat{\beta}` are computed using the estimated covariance matrix :math:`\widehat{\Omega}`:
 45
 46    .. math::
 47        \widehat{\operatorname{Var}}(\hat{g}) \approx A\widehat{\Omega}A^\top.
 48
 49    Pointwise and joint confidence intervals (via Gaussian multiplier bootstrap) are supported.

Best linear predictor (BLP) with orthogonal signals. Mainly used for CATE and GATE estimation for IRM models.

The Best Linear Predictor (BLP) targets the coefficient vector \( \beta_0 \) that minimizes the mean squared error between the true treatment effect function \( \tau(X) \) and a linear combination of basis functions \( b(X) \):

$$\beta_0 = \arg\min_{\beta \in \mathbb{R}^K} \mathbb{E}\Big[\big(\tau(X) - b(X)^\top \beta \big)^2\Big].$$

This is characterized by the moment condition:

$$\mathbb{E}[b(X)\psi] = \mathbb{E}[b(X)b(X)^\top]\beta_0,$$

where \( \psi \) is the orthogonal signal such that \( \mathbb{E}[\psi \mid X] = \tau(X) \).

The estimator is obtained via OLS of the orthogonal signal on the basis:

$$\hat{\beta} = (B^\top B)^{-1}B^\top\psi.$$

GATE (Group Average Treatment Effect)

When is_gate=True, the basis consists of group indicators (dummy variables). In this case, the BLP coefficients correspond to the group means of the orthogonal signal, which approximate the GATEs:

$$\hat{\beta}_k = \frac{1}{n_k}\sum_{i:G_i=k}\psi_i \approx \text{GATE}_k.$$

Confidence Intervals

Confidence intervals for any linear combination \( \hat{g} = A\hat{\beta} \) are computed using the estimated covariance matrix \( \widehat{\Omega} \):

$$\widehat{\operatorname{Var}}(\hat{g}) \approx A\widehat{\Omega}A^\top.$$

Pointwise and joint confidence intervals (via Gaussian multiplier bootstrap) are supported.
