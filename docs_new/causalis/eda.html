1275        def _silverman_bandwidth(x: np.ndarray) -> float:
1276            x = np.asarray(x, float)
1277            n = x.size
1278            if n < 2:
1279                return 0.04
1280            sd = np.std(x, ddof=1)
1281            iqr = np.subtract(*np.percentile(x, [75, 25]))
1282            s = sd if iqr <= 0 else min(sd, iqr / 1.34)
1283            h = 0.9 * s * n ** (-1 / 5)
1284            return float(max(h, 1e-6))

1286        def _kde_unbounded(x: np.ndarray, xs: np.ndarray, h: float) -> np.ndarray:
1287            """
1288            Gaussian KDE on R, implemented with NumPy (no SciPy).
1289            Handles degenerate cases by drawing a small bump at the mean.
1290            """
1291            x = np.asarray(x, float)
1292            if x.size == 0:
1293                return np.zeros_like(xs)
1294            if x.size < 2 or np.std(x) < 1e-12:
1295                mu = float(np.mean(x)) if x.size else 0.0
1296                h0 = max(h, 1e-3)
1297                z = (xs - mu) / h0

62        v_control = float(np.var(x[mask_c], ddof=1)) if np.sum(mask_c) > 1 else 0.0
63        v_treated = float(np.var(x[mask_t], ddof=1)) if np.sum(mask_t) > 1 else 0.0
64        pooled_std = float(np.sqrt((v_control + v_treated) / 2))
65        smd = float((mean_t_1 - mean_t_0) / pooled_std) if pooled_std > 0 else 0.0
66
67        # KolmogorovSmirnov test (works for continuous or dummy-discrete features)
68        try:
69            ks_res = ks_2samp(x[mask_t], x[mask_c])
70            ks_stat = float(ks_res.statistic)
71            ks_pvalue = float(ks_res.pvalue)
72        except Exception:
73            # Fallback to NaN if something unforeseen happens
74            ks_stat = np.nan
75            ks_pvalue = np.nan
76
77        rows.append(
78            {
79                "confounders": col,
80                "mean_t_0": mean_t_0,
81                "mean_t_1": mean_t_1,
82                "abs_diff": abs_diff,
83                "smd": smd,
84                "ks": ks_stat,
85                "ks_pvalue": ks_pvalue,
86            }
87        )
88
89    balance_df = pd.DataFrame(rows).set_index("confounders")
90
91    # Sort by absolute SMD value (most imbalanced first)
92    if "smd" in balance_df.columns:
93        balance_df = balance_df.reindex(balance_df["smd"].abs().sort_values(ascending=False).index)
94
95    return balance_df

Compute balance diagnostics for confounders between treatment groups.

Produces a DataFrame indexed by expanded confounder columns (after one-hot encoding categorical variables) with:

  • mean_t_0: mean value for control group (t=0)
  • mean_t_1: mean value for treated group (t=1)
  • abs_diff: abs(mean_t_1 - mean_t_0)
  • smd: standardized mean difference (Cohen's d using pooled std)
  • ks: KolmogorovSmirnov statistic between treated and control
  • ks_pvalue: p-value for the KS test
Parameters
  • df (pd.DataFrame): Source dataframe containing treatment and confounders.
  • treatment (str): Name of the treatment column (binary, 0/1 or False/True).
  • confounders (Iterable[str]): Column names of confounders to assess.
Returns
  • pd.DataFrame: Balance table sorted by |smd| (descending), index named 'confounders'.
